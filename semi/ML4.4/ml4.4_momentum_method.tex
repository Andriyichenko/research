\section{概要}
\thispagestyle{plain}
最急降下法（\text{Gradient Descent}）では、コスト関数の局所解（Local Minimum）に陥る可能性がある。これを回避し、かつ収束を加速させるために考案されたのが\textbf{モーメンタム法（Momentum Method）}である。
物理的な直感として、勾配降下法は「質点（ボール）が斜面を転がり落ちる運動」とみなせます。モーメンタム法では、このモデルに\textbf{速度（運動量）}と\textbf{摩擦（Friction）}の概念を導入する。

\begin{itemize}
    \item \textbf{速度（Velocity）}: 過去の勾配の値を積んで、慣性を持ってエネルギー障壁（Energy Barrier）を乗り越える助けとなる。
    \item \textbf{摩擦（Friction）}: システムのエネルギーを徐々に散逸させ、ボールが最終的に平衡点（最小値）に止まるようにする。
\end{itemize}

\textbf{Polyak (1964)} による古典的なモーメンタム法の更新式は以下の通りである。
\begin{empheq}{align}
    x^{n+1} &= x^n + v^{n+1} \label{eq:pos_update_polyak} \\
    v^{n+1} &= \mu v^n - \eta \nabla f(x^n) \label{eq:vel_update_polyak}
\end{empheq}
ここで、$\eta > 0$ は学習率、$\mu \in (0, 1]$ はモーメンタム係数（摩擦係数とも呼ばれる）である。

\section{Kinematic Interpretation\\ \quad \ \ 運動学的解釈}

目的関数 $f(x)$ によって形成される「器（cup）」の中を転がる質量 $m=1$ のボールを考える。
ボールには以下の2つの力が作用する。
\begin{enumerate}
    \item \textbf{勾配力}: $-\nabla f(x)$ （重力のようにポテンシャルを下る力）
    \item \textbf{摩擦力}: $F_f = -\footnotemark[1]\rho \dot{x}(t)$ （速度に比例し、運動を妨げる力。$\rho > 0$ は減衰係数）
\end{enumerate}
\footnotetext[1]{$\rho$ はローと言う}
\noindent
よって、$F_{\textbf{合力}} = F_{\textbf{勾配力}} + F_{\textbf{摩擦力}} = - \rho \dot{x}(t) -\nabla f(x)$。\\
ニュートンの運動方程式（$\mathbf{F=ma}$）より、以下の微分方程式が得られる。

\begin{equation}
    a =\footnotemark[2]\ddot{x}(t) = -\rho \dot{x}(t) - \nabla f(x(t)) \label{eq:newton_friction}
\end{equation}
\footnotetext[2]{$\dot{v} = \frac{d}{dt}v = \frac{d}{dt}\left(\frac{d}{dt}x\right) = \frac{d^2}{dt^2}x = \ddot{x} = -\nabla f(x)$. ここで、$\ddot{x}$は加速度であり、\textbf{エックス・ツー・ドット}と呼ぶ。}
\subsection{Energy Dissipation Analysis\\ エネルギー散逸の解析}

\begin{reidai}{$\frac{d}{dt} E_{tot}(t)< 0$の証明}{dd}
  このシステムにおいて、総エネルギー $E_{tot}(t)$ が保存されず、時間とともに減少することを証明する。
\end{reidai}
\vspace{0.5em}
\begin{proof}
 \ \\
\noindent
総エネルギーは運動エネルギーとポテンシャルエネルギーの和で定義される。
\[
E_{tot}(t) = \frac{1}{2}\|\dot{x}(t)\|^2 + f(x(t))
\]
これを時間 $t$ で微分する。
\begin{align*}
    \frac{d}{dt} E_{tot}(t) &= \frac{d}{dt} \left( \frac{1}{2}\dot{x}(t)^T \dot{x}(t) + f(x(t)) \right) \\
    &= \dfrac{1}{2}((\dfrac{d}{dt}\dot{x}(t))^T \dot{x}(t) + \dot{x}(t)^T \dfrac{d}{dt}\dot{x}(t)) + \nabla f(x(t))^T \dfrac{d}{dt}(x(t)) \\
    &= \dot{x}(t)^T \ddot{x}(t) + \dot{x}(t)^T \nabla f(x(t))  \\
    &= \dot{x}(t)^T \left( \ddot{x}(t) + \nabla f(x(t)) \right)
\end{align*}
ここで、運動方程式 \eqref{eq:newton_friction} より、$\ddot{x}(t) + \nabla f(x(t)) = -\rho \dot{x}(t)$ なので、これを代入する。
\begin{align*}
\frac{d}{dt} E_{tot}(t) &= \dot{x}(t)^T \left( -\rho \dot{x}(t) \right) \\
&= -\rho \|\dot{x}(t)\|^2 \\
&= -\rho \|v(t)\|^2 < 0
\end{align*}
$\rho > 0$ であるため、速度がゼロでない限りエネルギーは常に減少する。これにより、システムはエネルギーの低い状態（平衡点）へと収束する。
\end{proof}

\subsection{Phase Space Analysis and Divergence\\ 相空間解析と発散率div}
運動方程式 \eqref{eq:newton_friction} は、位置 $x$ と速度 $v$ に関する1階連立微分方程式\textbf{ODEs}として記述できる。
\[
\begin{cases}
    \dot{x}(t) = v(t) \\
    \dot{v}(t) = -\rho v(t) - \nabla f(x(t))
\end{cases}
\]
このベクトル場 $X = (\dot{x}, \dot{v})$ の発散（\textbf{Divergence}）を計算する。

\begin{align*}
\text{div}(\dot{x}, \dot{v}) &= \frac{\partial}{\partial x}(\dot{x}) + \frac{\partial}{\partial v}(\dot{v}) \\
&= \frac{\partial}{\partial x}(v) + \frac{\partial}{\partial v}(-\rho v - \nabla f(x)) \\
&= 0 + (-\rho) \\
&= -\rho < 0
\end{align*}
発散が負であることは、相空間内の体積が時間とともに\textbf{収縮（contracting）}することを意味する。これにより、解軌道は一点（平衡点）に収束する。

\section{Discretization and Algorithm Derivation\\ \quad \ \ 離散化とアルゴリズムの導出}

コンピュータ上で計算を実行するために、連続時間の\textbf{ODEs}システムを有限差分法を用いて離散化する。\\

\noindent
まず、等間隔の時間分割 $0 = t_0 < t_1 < \dots < t_n < \infty$ を考える。時間刻み幅を $\Delta t = t_{n+1} - t_n$ \textbf{（定数）}とする。
システムの$n$ ステップ目の状態を $(x^n, v^n) = (x(t_n), v(t_n))$ と表記する。

微分 $\dot{x}(t), \dot{v}(t)$ を前進差分（Forward Difference）で近似する
\[
\dot{x}(t) \approx \frac{x^{n+1} - x^n}{\Delta t}, \quad \dot{v}(t) \approx \frac{v^{n+1} - v^n}{\Delta t}
\]
これらを元の\textbf{ODEs}に代入すると
\begin{align*}
    \frac{x^{n+1} - x^n}{\Delta t} &= v^n \\
    \frac{v^{n+1} - v^n}{\Delta t} &= -\rho v^n - \nabla f(x^n)
\end{align*}
となり、これを整理すると、以下の差分方程式が得られる。
\begin{empheq}[left=\empheqlbrace]{align}
    x^{n+1} - x^n &= v^n \Delta t \label{eq:diff_x} \\
    v^{n+1} - v^n &= -\rho v^n \Delta t - \nabla f(x^n) \Delta t \label{eq:diff_v}
\end{empheq}

\noindent
次に、物理のパラメータをアルゴリズムのハイパーパラメータに変換する。
\begin{itemize}
    \item 時間刻みを $\epsilon = \Delta t$ と置く。
    \item 摩擦項をまとめるため、$\mu = 1 - \rho \epsilon$ と定義する。ここで $\rho > 0, \epsilon > 0$ より $\mu < 1$ である。
\end{itemize}
式 \eqref{eq:diff_v} の右辺を変形する：
\[
v^{n+1} = v^n - \rho v^n \Delta t - \nabla f(x^n) \Delta t = (1 - \rho \Delta t)v^n - \Delta t \nabla f(x^n)
\]
パラメータ $\epsilon, \mu$ を代入すると、中間的なシステムが得られる。
\begin{align}
    x^{n+1} &= x^n + \epsilon v^n \label{eq:pos_inter} \\
    v^{n+1} &= \mu v^n - \epsilon \nabla f(x^n) \label{eq:vel_inter}
\end{align}

\noindent
最後、物理的な速度 $v$ とアルゴリズム上の更新量を整合させるため、速度変数の再スケーリング（Rescaling）を行う。
新しい速度変数 $\tilde{v}$ を以下のように定義する。
\[
\tilde{v}^n = \epsilon v^n \quad \left(\iff v^n = \frac{\tilde{v}^n}{\epsilon}\right)
\]
これを式 \eqref{eq:pos_inter} と \eqref{eq:vel_inter} に代入して計算を進める

\textbf{位置の更新式:}
\[
x^{n+1} = x^n + \epsilon \left( \frac{\tilde{v}^n}{\epsilon} \right) = x^n + \tilde{v}^n
\]

\textbf{速度の更新式:}
\[
\frac{\tilde{v}^{n+1}}{\epsilon} = \mu \left( \frac{\tilde{v}^n}{\epsilon} \right) - \epsilon \nabla f(x^n)
\]
この両辺に $\epsilon$ を掛けると、
\[
\tilde{v}^{n+1} = \mu \tilde{v}^n - \epsilon^2 \nabla f(x^n)
\]
ここで、学習率を $\eta = \epsilon^2$ とすると、最終的なアルゴリズムの更新式が得られる。
\begin{empheq}{align}
    x^{n+1} &= x^n + \tilde{v}^n \label{eq:finalx}\\
    \tilde{v}^{n+1} &= \mu \tilde{v}^n - \eta \nabla f(x^n)\label{eq:finalv}
\end{empheq}
これはPolyakの古典的なモーメンタム法（式 [\ref{eq:pos_update_polyak}], [\ref{eq:vel_update_polyak}]）と形式的に一致する。

\section{Analysis of Example 4.4.1\\ \quad \ \ 例4.4.1の解析}

2次関数に対するモーメンタム法の応用を、線形代数で解析する。

\begin{reidai}{Quadratic Function Analysis}{ex441}
    実変数 $x$ の2次関数 $f(x) = \frac{1}{2}(ax - b)^2$ （ただし $a \ne 0$）を考える。
    このとき、モーメンタム法の更新式 \eqref{eq:finalx}-\eqref{eq:finalv}は線形式 $s_{n+1} = M s_n + \beta$ として記述できる。
    ここで、$s_n = \begin{pmatrix} x^n \\ v^n \end{pmatrix}, \quad M = \begin{pmatrix} 1 & \epsilon \\ -\epsilon a^2 & \mu \end{pmatrix}, \quad \beta = \begin{pmatrix} 0 \\ \epsilon ab \end{pmatrix}$。
    \begin{enumerate}
        \item 状態 $s_n$ の一般項を初期状態 $s_0$ を用いて導出せよ。
        \item 行列 $M$ の固有値を解析し、システムが平衡点へ収束することを証明せよ。
    \end{enumerate}
\end{reidai}
\vspace{0.5em}
\begin{proof}
 \ \\
\noindent
まず、勾配 $f'(x) = a(ax - b) = a^2 x - ab$ を更新式に代入し、行列形式に整理する。
\begin{align*}
    x^{n+1} &= x^n + \epsilon v^n \\
    v^{n+1} &= \mu v^n - \epsilon (a^2 x^n - ab) = -\epsilon a^2 x^n + \mu v^n + \epsilon ab
\end{align*}
状態ベクトルを $s_n = \begin{pmatrix} x^n \\ v^n \end{pmatrix}$ と置くと、以下の線形漸化式が得られる。
\begin{equation}
    s_{n+1} = M s_n + \beta, \quad \text{where,\  } 
    M = \begin{pmatrix} 1 & \epsilon \\ -\epsilon a^2 & \mu \end{pmatrix}, \quad 
    \beta = \begin{pmatrix} 0 \\ \epsilon ab \end{pmatrix}
    \label{eq:matrix_system}
\end{equation}


次、漸化式 \eqref{eq:matrix_system} を繰り返し計算し、一般項 $s_n$ を導出する。

\begin{itemize}
    \item $n=1$:
    \[ s_1 = M s_0 + \beta \]
    \item $n=2$:
    \[ s_2 = M s_1 + \beta = M(M s_0 + \beta) + \beta = M^2 s_0 + M\beta + \beta \]
    \item $n=3$:
    \[ s_3 = M s_2 + \beta = M(M^2 s_0 + M\beta + \beta) + \beta = M^3 s_0 + (M^2 + M + I)\beta \]
\end{itemize}

よって、これを $n$ 回繰り返すと、以下の形式となる。
\begin{equation}
    s_n = M^n s_0 + \left( \sum_{k=0}^{n-1} M^k \right) \beta
\end{equation}
ここで、括弧内の項は行列の等比級数である。$S_n = \displaystyle\sum_{k=0}^{n-1} M^k$ と置き、等式の両辺に$I - M$を掛けると、
\[
(\mathbb{I} - M) S_n = (\mathbb{I} - M)(\mathbb{I} + M + \dots + M^{n-1}) = \mathbb{I} - M^n
\]
ここで、$\mathbb{I}$は単位行列(\textbf{Unit Matrix})行列 $\mathbb{I}-M$ が正則（逆行列が存在）であると仮定すると、
\[ \footnotemark[3]S_n = (\mathbb{I} - M^n)(\mathbb{I} - M)^{-1} \]
したがって、状態 $s_n$ の一般項は以下のように表される。
\footnotetext[3]{具体的な証明は\textbf{\ref{app:inverse}}を参照してください}
\begin{empheq}{equation}
    s_n = M^n s_0 + (\footnotemark[4]\mathbb{I}_2 - M^n)(\mathbb{I}_2 - M)^{-1} \beta \label{eq:closed_form}
\end{empheq}
\footnotetext[4]{$\mathbb{I}_2$ とは$2\times 2$の単位行列である。}
あと、収束先を求めるために必要な $(\mathbb{I}_2 - M)^{-1}$ を計算する。
\[
\mathbb{I}_2 - M = \begin{pmatrix} 1 & 0 \\ 0 & 1 \end{pmatrix} - \begin{pmatrix} 1 & \epsilon \\ -\epsilon a^2 & \mu \end{pmatrix} 
= \begin{pmatrix} 0 & -\epsilon \\ \epsilon a^2 & 1-\mu \end{pmatrix}
\]
\textbf{2x2}行列の逆行列公式より、行列式は $\det = 0 - (-\epsilon)(\epsilon a^2) = \epsilon^2 a^2$ なので、
\begin{equation}
    (\mathbb{I}_2 - M)^{-1} = \frac{1}{\epsilon^2 a^2} \begin{pmatrix} 1-\mu & \epsilon \\ -\epsilon a^2 & 0 \end{pmatrix} \label{eq:inverse_m}
\end{equation}
となる。\\

\noindent
このシステムが収束させるためには、$\displaystyle\lim_{n \to \infty} M^n = 0$ である必要がある。\ie $M$ のすべての固有値の絶対値が1未満であることと同値である。\\

\noindent
次に、行列 $M$ の特性方程式（\textbf{Characteristic Equation}）を解く
\[
\det(M - \lambda \mathbb{I}) = \det \begin{pmatrix} 1-\lambda & \epsilon \\ -\epsilon a^2 & \mu-\lambda \end{pmatrix} = 0
\]
\[
(1-\lambda)(\mu-\lambda) + \epsilon^2 a^2 = 0 \implies \lambda^2 - (1+\mu)\lambda + (\mu + \epsilon^2 a^2) = 0
\]
この2次方程式の解 $\lambda_1, \lambda_2$ について分析する（\textbf{7ページの図\ref{fig:2}参照}）
\begin{enumerate}
    \item $\epsilon = 0$ のとき：解は $\lambda_1(0)=\mu$ と $\lambda_2(0)=1$ である。ここで $0 < \mu < 1$ である。
    \item $\epsilon > 0$ （$\mathtt{for\ small\ enough}$）のとき：定数項が増加し、放物線が上方にシフトする。$\epsilon$ が十分小さければ、連続性により解は実数のまま区間 $(\mu, 1)$ の内部にある。
\end{enumerate}
よって、$0 < \lambda_1(\epsilon) < \lambda_2(\epsilon) < 1$ が成立するため、
\[ \lim_{n \to \infty} M^n = \mathbf{O_{2 \times 2}} \quad (\text{ゼロ行列}) \]
が証明された。

\noindent
極限 $n \to \infty$ を式 \eqref{eq:closed_form} に適用する。
第一項 $M^n s_0$ は 0 に収束し、第二項の $M^n$ も 0 になる。
\begin{align*}
    s^* = \lim_{n \to \infty} s_n &= (\mathbb{I}_2 - O)(\mathbb{I}_2 - M)^{-1} \beta \\
    &= (\mathbb{I}_2 - M)^{-1} \beta
\end{align*}
式 \eqref{eq:inverse_m} の結果と $\beta = (0, \epsilon ab)^T$ を代入して計算する。
\[
s^* = \frac{1}{\epsilon^2 a^2} \begin{pmatrix} 1-\mu & \epsilon \\ -\epsilon a^2 & 0 \end{pmatrix} \begin{pmatrix} 0 \\ \epsilon ab \end{pmatrix}
\]
行列の積を計算すると：
\begin{itemize}
    \item $x$成分: $\frac{1}{\epsilon^2 a^2} \left( (1-\mu)\cdot 0 + \epsilon \cdot \epsilon ab \right) = \frac{\epsilon^2 ab}{\epsilon^2 a^2} = \frac{b}{a}$
    \item $v$成分: $\frac{1}{\epsilon^2 a^2} \left( (-\epsilon a^2)\cdot 0 + 0 \cdot \epsilon ab \right) = 0$
\end{itemize}
となる。よって、
\[
s^* = \begin{pmatrix} b/a \\ 0 \end{pmatrix}
\]
これは $f(x)$ の最小値を与える点 $x^* = b/a$ （勾配 $a(ax - b)=0$ の解）と完全に一致する。
以上より、モーメンタム法は初期値に関わらず正しい最適解へ収束することが示された。

\noindent
\textbf{モーメンタム法の命名について}:
通常、物理学における「運動量（Momentum）」は物体を前へ押し進めるものであるが、この手法では実際には摩擦力（$\mu < 1$）を用いて粒子を\textbf{減衰（Damp）}させている。これは平衡点を行き過ぎる（\textbf{Overshoot}）のを防ぐためである。
もし局所（\textbf{local}）解から脱出するために「勢い」をつけたい場合は、$\mu > 1$ （負の摩擦＝加速）を設定する必要がある。

\section{Convergence Conditions\\ \quad \ \ 収束条件}

モーメンタム法によって生成される数列 $(x^n)_n$ および $(v^n)_n$ の収束性を議論するために、まずこれらの数列の閉じた形式（Closed-form expression）、すなわち一般項を導出する。

\subsection{Exact Formulas for Sequences\\ 数列の厳密な公式}

位置 $x^n$ と速度 $v^n$ の漸化式を繰り返し展開（Iterating）することで、一般項を求める。

\paragraph{位置 $x^n$ の導出}
式 \eqref{eq:finalx} $x^n = x^{n-1} + v^n$ を繰り返し適用する。
\begin{align*}
    x^n &= x^{n-1} + v^n \\
    x^{n-1} &= x^{n-2} + v^{n-1} \\
    &\vdots \\
    x^1 &= x^0 + v^1
\end{align*}
これらを辺々加えると、中間項がなくされ、以下の式が得られる。
\begin{empheq}{equation}
    x^n = x^0 + \sum_{k=1}^{n} v^k \label{eq:xn_closed}
\end{empheq}

\paragraph{速度 $v^n$ の導出}
簡単のため、$b_n = \nabla f(x^n)$ と置く。式 \eqref{eq:finalv} $v^n = \mu v^{n-1} - \eta b_{n-1}$ を展開する。
\begin{align*}
    v^n &= \mu v^{n-1} - \eta b_{n-1} \\
    &= \mu (\mu v^{n-2} - \eta b_{n-2}) - \eta b_{n-1} \quad \cdots (\text{where,\ } v^{n-1} = \mu v^{n-2} - \eta b_{n-2})\\
    &= \mu^2 v^{n-2} - \mu \eta b_{n-2} - \eta b_{n-1} \\
    &= \mu^2 (\mu v^{n-3} - \eta b_{n-3}) - \mu \eta b_{n-2} - \eta b_{n-1} \\
    &= \mu^3 v^{n-3} - \eta (\mu^2 b_{n-3} + \mu b_{n-2} + b_{n-1})
\end{align*}
このように繰り返すと、
\[
v^n = \mu^n v^0 - \eta \sum_{j=0}^{n-1} \mu^{n-1-j} b_j
\]
インデックスを整理（$v^{n+1}$ の形にし、和の添字を調整）すると、以下の形式が得られる。
\begin{empheq}{equation}
    v^{n+1} = \mu^{n+1} v^0 - \eta \sum_{i=0}^{n} \mu^{n-i} b_i \label{eq:vn_closed}
\end{empheq}
この右辺第2項は、勾配$b_i$ と減衰係数 $\mu$ のべき乗との\textbf{畳み込み（Convolution）}の形になっている。

\subsection{Mathematical Tools for Convergence\\ 収束解析のための数学的道具}

$v^{n+1}$ を解析するために、次の畳み込み級数に関する定義と定理を導入する。

\begin{dfn}[Convolution Series]\label{dfn:convolution_series}\ \\
    2つの数列 $(a_n)_{n \geq 0}$ と $(b_n)_{n \geq 0}$ に対し、その畳み込み級数 $(c_n)_{n \geq 0}$ は以下のように定義される。
    \[
    c_n = \sum_{i=0}^{n} a_i b_{n-i}
    \]
    また、無限級数としての畳み込みは $\sum_{n \geq 0} c_n$ で表される。
\end{dfn}

\begin{prop}[Limit of Convolution Term]\footnotemark[5]\label{prop:limit}\ \\
    実数列 $(a_n)$ が $0$ に収束し、級数 $\sum b_n$ が絶対収束すると仮定する。このとき、畳み込み項の極限は $0$ になる。
    \[
    \lim_{n \to \infty} \sum_{i=0}^{n} a_i b_{n-i} = 0
    \]
\end{prop}
\footnotetext[5]{畳み込み級数の極限については、\textbf{Appendix\ B.\ \ref{app:5.2}} を参照してください。}
\begin{thm}[Convergence of Convolution Series]\footnotemark[6]\label{thm:convergence}\ \\
    級数 $\sum a_n$ が収束し、級数 $\sum b_n$ が絶対収束するとする。このとき、それらの畳み込み級数 $\sum c_n$ も収束し、その和は各級数の和の積に等しい。
    \[
    \sum_{n \ge 0} \left( \sum_{i=0}^{n} a_i b_{n-i} \right) = \left( \sum_{n \ge 0} a_n \right) \left( \sum_{n \ge 0} b_n \right)
    \]
\end{thm}
\footnotetext[6]{畳み込み級数の収束については、\textbf{Appendix\ B.\ \ref{app:5.3}} を参照してください。}
\subsection{Main Convergence Results\\ 主要な収束結果}

以上の道具を用いて、モーメンタム法の収束条件を記述する以下の重要な命題を証明する。

\begin{prop}[Convergence of Momentum Method]\ \\
    \begin{enumerate}[label=(\alph*)]
        \item もし勾配 $\nabla f(x^n)$ が $0$ に収束するならば、速度列 $(v^n)$ も $n \to \infty$ で $0$ に収束する。
        \item もし勾配のノルムの級数 $\sum_{n \ge 0} \|\nabla f(x^n)\|$ が収束するならば、数列 $(x^n)$ および $(v^n)$ は共に収束する。
    \end{enumerate}
\end{prop}


\paragraph{(a) の証明}

\begin{proof}\ \\

\noindent
式 \eqref{eq:vn_closed} を考える。
\[
v^{n+1} = \mu^{n+1} v^0 - \eta \sum_{i=0}^{n} \mu^{n-i} b_i
\]
ここで $0 < \mu < 1$ であるため、$\mu^n$ は $n \to \infty$ で $0$ に収束する。よって$\mu^{n+1} v^0 = 0$。\\
第2項について：
\begin{itemize}
    \item 数列 $\mu^n$ は幾何級数であり、$\sum |\mu^n|$ は絶対収束する。
    \item 仮定より $b_n = \nabla f(x^n) \to 0$ である。
\end{itemize}
これらを \textbf{\texttt{Prop.}\ref{prop:limit}} に適用（$a_i$を $b_i$、$b_{n-i}$を$\mu^{n-i}$とみなす）すると、畳み込み項の極限は0となる。
したがって、
\[
v^* = \lim_{n \to \infty} v^{n+1} =  v^0 \cdot 0 - \eta \cdot 0 = 0
\]
\end{proof}

\paragraph{(b) の証明}
\begin{proof}\ \\

\noindent
仮定より $\sum \|\nabla f(x^n)\|$ が収束している。級数が収束するための必要条件は一般項が0に収束することであるため\footnotemark[7]、$\|\nabla f(x^n)\| \to 0$、\ie $\nabla f(x^n) \to 0$。
よって、(a) の結果より直ちに \textbf{$v^n$ は 0 に収束する}。\\

% 本文中の該当箇所
\footnotetext[7]{級数が収束するための必要条件（一般項が0に収束する）の証明については、\textbf{\ref{app:nec_condition}} を参照してください。}


\noindent
次に $x^n$ の収束性を示す。式 \eqref{eq:xn_closed} より $x^{n+1} = x^0 + \sum_{k=0}^n v^{k+1}$ であるから、級数 $\sum v^{k+1}$ の収束を示せば良い。
式 \eqref{eq:vn_closed} を代入して整理する。
\begin{align*}
    x^{n+1} &= x^0 + \sum_{k=0}^{n} \left( \mu^{k+1} v^0 - \eta \sum_{i=0}^{k} \mu^{k-i} b_i \right) \\
    &= x^0 + v^0 \sum_{k=0}^{n} \mu^{k+1} - \eta \sum_{k=0}^{n} \left( \sum_{i=0}^{k} \mu^{k-i} b_i \right)
\end{align*}
ここで $n \to \infty$ の極限を取る。
\begin{itemize}
    \item 第2項：等比級数の和 $\sum \mu^{k+1}$ は収束する（和は $\frac{\mu}{1-\mu}$）。
    \item 第3項：$\sum \mu^n$ は絶対収束し、$\sum \|b_n\|$ も収束（仮定より）するため、$\sum b_n$ も絶対収束する。よって \textbf{\texttt{Thm.}\ref{thm:convergence}} より、この畳み込み級数も収束する。
\end{itemize}
したがって、極限 $x^* = \displaystyle\lim_{n \to \infty} x^{n+1}$ が存在し、以下のように書ける。
\[
x^* = x^0 + v^0 \frac{\mu}{1-\mu} - \frac{\eta}{1-\mu} \sum_{n \ge 0} \nabla f(x^n)
\]
$x^*$は閉じた形式の解があることにより、$x^n$ も収束することが示された。
\end{proof}

\section{Nesterov Accelerated Gradient (NAG)}

\begin{rem}{Nesterov Accelerated Gradient}{nag}
    標準的なモーメンタム法の改良版として、Nesterov (1983) によって提案された \textbf{NAG} がある。
    これは勾配を計算する位置を変更する手法である。
    現在の位置 $x^n$ で勾配を計算する代わりに、慣性項によって「次に到達するであろう位置」$x^n + \mu v^n$ で勾配を評価する。
    \begin{empheq}{align}
        x^{n+1} &= x^n + v^{n+1} \label{eq:nag_x} \\
        v^{n+1} &= \mu v^n - \eta \nabla f(x^n + \mu v^n) \label{eq:nag_v}
    \end{empheq}
    この「先読み（Look-ahead）」により、NAGは振動を抑制し、収束率を向上させることができる。
\end{rem}

以下に、NAGのアルゴリズム的な詳細と、なぜこの手法が標準的なモーメンタム法より優れているのか、その\textbf{更新ロジック}と\textbf{その改善点}に分けて分析する。

\subsection{Update Logic of NAG\\ \ NAGの更新ロジック}

数式 \eqref{eq:nag_x} と \eqref{eq:nag_v} は理論的に美しいが、実装においては「勾配を計算する点」と「実際に更新する点」が異なるため、手順を明確に理解する必要がある。NAGは以下の2段階のプロセスと解釈できる。

\begin{enumerate}[label=\textbf{Step \arabic*.}]
    \item \textbf{予測（Prediction）}: 
    まず、勾配を無視して、慣性（Momentum）だけで移動した場合の仮の位置 $\hat{x}^n$ を計算する。
    \[ \hat{x}^n = x^n + \mu v^n \]
    \item \textbf{修正（Correction）}: 
    この「先読みした位置 $\hat{x}^n$」における勾配 $\nabla f(\hat{x}^n)$ を計算し、それを用いて現在の速度ベクトルを修正し、最終的な位置更新を行う。
    \[ v^{n+1} = \mu v^n - \eta \nabla f(\hat{x}^n) \]
    \[ x^{n+1} = x^n + v^{n+1} \]
\end{enumerate}

このロジックを直感的に言えば、\textbf{「盲目のハイカー」と「先を見通すハイカー」の違い}である。
標準的なMomentum法は、現在の斜度だけを見て勢いよく飛び出すが、NAGは「このまま進んだらどうなるか」を一旦確認してから、足の踏み出し方を微調整するのである。

\subsection{Major Improvements\\ 主要な改善点}

NAGが標準的な勾配降下法（Gradient Descent, GD）やMomentum法よりも優れている理由は、以下である。

\subsubsection{Correction of Overshooting\\ Overshoot（行き過ぎ）の抑制}

谷底（最適解）に向かってボールが転がり落ちる状況を想像してほしい。
\begin{itemize}
    \item \textbf{Standard Momentum}: 谷底に近づいても、過去の加速（$\mu v^n$）が残っているため、勢い余って谷底を通り過ぎてしまい、反対側の斜面を登ってしまう（オーバーシュート）。現在の位置での勾配は減速を指示するかもしれないが、過去の勢いが勝る場合がある。
    \item \textbf{NAG (Look-ahead)}: NAGは「もし慣性で進んだら、谷底を通り過ぎて反対側の斜面（上り坂）に達する」ことを\textbf{移動する前に}検知する\footnotemark[8]。
    先読み点での勾配 $\nabla f(x^n + \mu v^n)$ は「戻れ（逆方向）」という情報を指しているため、速度更新式において慣性項 $\mu v^n$ を強力に打ち消すブレーキとして作用する。
\end{itemize}
\footnotetext[8]{この先読み効果により、パラメータの更新ベクトルは常に「曲率（Curvature）」の情報を間接的に取り込むことになる。}

\section{AdaGrad}
AdaGrad（Adaptive Gradient）という適応的学習率を持つ修正確率的勾配降下法は、2011年にDuchiらによって発表されました。最小化の対象となるコスト関数を $C(x)$（ただし $x \in \mathbb{R}^N$）とし、ステップ $t$ で評価された勾配ベクトルを $g_t = \nabla C(x(t))$ と表記します。

ここで、過去の勾配の情報を蓄積する $N \times N$ 行列 $G_t$ を次のように定義します：
\begin{align*}
    G_t = \sum_{\tau=1}^t g_\tau g_\tau^T
\end{align*}
この行列を用いたパラメータの更新式は以下のようになります：
\begin{align*}
    x(t+1) = x(t) - \eta G_t^{-1/2} g_t
\end{align*}
ここで $\eta > 0$ は学習率です。離散的な時間ステップにおいては、$x^{n+1} = x^n - \eta G_n^{-1/2} g_n$ と等価に記述できます。

\hosoku \textbf{行列の対角化近似}

高次元のパラメータ空間（$N$ が数百万〜数十億）において、完全な $N \times N$ 行列 $G_t$ の逆平方根 $G_t^{-1/2}$ を計算することは、計算量（$\mathcal{O}(N^3)$）およびメモリの観点から完全に非現実的です。この計算の爆発を避けるため、AdaGradの更新式では非対角要素（異なるパラメータ間の勾配の相関）を $0$ とみなす巧妙な近似が行われます。

\paragraph{対角行列 $\operatorname{diag}(G_t)$ の構築}\ \\
勾配ベクトルの直積 $g_\tau g_\tau^T$ の全要素は以下のように展開されます：
\begin{align*}
    g_\tau g_\tau^T = 
    \begin{pmatrix} 
    (g_{\tau,1})^2 & g_{\tau,1}g_{\tau,2} & \dots & g_{\tau,1}g_{\tau,N} \\ 
    g_{\tau,2}g_{\tau,1} & (g_{\tau,2})^2 & \dots & g_{\tau,2}g_{\tau,N} \\ 
    \vdots & \vdots & \ddots & \vdots \\ 
    g_{\tau,N}g_{\tau,1} & g_{\tau,N}g_{\tau,2} & \dots & (g_{\tau,N})^2 
    \end{pmatrix}
\end{align*}
ここから非対角要素 $g_{\tau,i}g_{\tau,j} \ (i \neq j)$ を捨てて対角成分のみを抽出し、時間 $\tau=1$ から $t$ まで足し合わせると、非対角要素が全て $0$ である対角行列が得られます：
\begin{align*}
    \operatorname{diag}(G_t) = 
    \begin{pmatrix} 
    \sum_{\tau=1}^t (g_{\tau,1})^2 & 0 & \dots & 0 \\ 
    0 & \sum_{\tau=1}^t (g_{\tau,2})^2 & \dots & 0 \\ 
    \vdots & \vdots & \ddots & \vdots \\ 
    0 & 0 & \dots & \sum_{\tau=1}^t (g_{\tau,N})^2 
    \end{pmatrix}
\end{align*}

\paragraph{対角行列のべき乗の性質の利用}\ \\
対角行列の性質として、「行列の逆行列や平方根は、主対角線上にある各要素に個別にその演算（逆数や平方根）を適用するだけで求まる」という点があります。そのため、複雑な行列の逆平方根演算を行うことなく、直ちに次のように記述できます：
\begin{align*}
    \operatorname{diag}(G_t)^{-1/2} = 
    \begin{pmatrix} 
    \frac{1}{\sqrt{\sum (g_{\tau,1})^2}} & 0 & \dots & 0 \\ 
    0 & \frac{1}{\sqrt{\sum (g_{\tau,2})^2}} & \dots & 0 \\ 
    \vdots & \vdots & \ddots & \vdots \\ 
    0 & 0 & \dots & \frac{1}{\sqrt{\sum (g_{\tau,N})^2}} 
    \end{pmatrix}
\end{align*}

\paragraph{「行列 $\times$ ベクトル」から「スカラーの要素ごと演算」への帰着}\ \\
最後に、この行列 $\operatorname{diag}(G_t)^{-1/2}$ を現在の勾配ベクトル $g_t = (g_{t,1}, g_{t,2}, \dots, g_{t,N})^T$ に掛けます。対角行列と列ベクトルの積は、実質的に「対応する行の要素同士の掛け算」へと崩壊します。
    
したがって、ベクトル全体の更新式 $x(t+1) = x(t) - \eta \operatorname{diag}(G_t)^{-1/2} g_t$ から、第 $j$ 次元目のパラメータ $x_j$ の独立した更新式を抽出すると、次のような極めてシンプルな形に帰着します：
\begin{align*}
    x_j(t+1) = x_j(t) - \frac{\eta}{\sqrt{\sum_{\tau=1}^t (g_{\tau,j})^2}} g_{t,j}
\end{align*}
\paragraph{物理的な意味と結論：}\ \\
このように非対角要素を捨てる近似を行うことで、計算量は $\mathcal{O}(N^3)$ から $\mathcal{O}(N)$ に激減します。同時に、「各パラメータの更新は他の次元と完全に独立しており、自分自身の過去の勾配の2乗和が大きい（＝過去によく更新され、傾斜が急であった）パラメータほど、現在のステップ幅が小さくなるようにペナルティが与えられる」という、各次元に独立した適応的な学習率のメカニズムが見事に実現されています。

\begin{reidai}{2次元座標系におけるAdaGradの勾配方向補正のシミュレーション}{adagrad_sim}
極端に急な方向と緩やかな方向を持つ「渓谷」のようなコスト関数を考えます：
\begin{align*}
    C(x_1, x_2) = 50x_1^2 + \frac{1}{2}x_2^2
\end{align*}
この関数の大域的最小値は原点 $(0, 0)$ です。$x_1$ 軸方向は係数が50であり非常に急勾配ですが、$x_2$ 軸方向は係数が0.5であり非常に緩やかです。

初期位置を $x(1) = (1, 1)^T$、学習率を $\eta = 0.1$ と設定し、標準の勾配降下法（Standard GD）とAdaGradの1ステップ目の更新ベクトルを比較することで、行列 $G_t^{-1/2}$ がどのように勾配の方向を補正するのかをシミュレーションする。
\end{reidai}
\begin{proof}
まず、コスト関数の勾配（導関数）ベクトルは以下のようになります：
\begin{align*}
    g_t = \nabla C = \begin{pmatrix} 100x_1 \\ x_2 \end{pmatrix}
\end{align*}
初期位置 $x(1) = (1, 1)^T$ における真の勾配は $g_1 = (100, 1)^T$ です。

\paragraph{標準的な勾配降下法 (Standard GD) の軌跡}\ \\
基礎的な勾配降下法の更新式は $\Delta x = -\eta g_1$ です。数値を代入すると：
\begin{align*}
    \Delta x_{GD} = -0.1 \begin{pmatrix} 100 \\ 1 \end{pmatrix} = \begin{pmatrix} -10 \\ -0.1 \end{pmatrix}
\end{align*}
\textit{考察：} 更新ベクトルの方向は $(-10, -0.1)^T$ です。これは、急な $x_1$ 軸方向に10も進んでしまい（原点を飛び越えて$-9$に達し、激しい振動を引き起こします）、緩やかな $x_2$ 軸方向にはわずか0.1しか進まないことを意味します。2次元座標系で描くと、このベクトルはほぼ水平な線となり、最適解 $(0,0)$ へ向かう最短経路から完全に逸脱しています。

\paragraph{AdaGrad による行列変換の魔法}\ \\
AdaGradでは、$G_t^{-1/2}$ 行列の対角近似 $\operatorname{diag}(G_t)$ を用いてこの問題を解決します。

\textit{ステップ1：過去の勾配の蓄積行列 $\operatorname{diag}(G_1)$ の計算}\\
第1ステップであるため、累積和は現在の勾配の2乗のみで構成されます：
\begin{align*}
    \operatorname{diag}(G_1) = \begin{pmatrix} 100^2 & 0 \\ 0 & 1^2 \end{pmatrix} = \begin{pmatrix} 10000 & 0 \\ 0 & 1 \end{pmatrix}
\end{align*}

\textit{ステップ2：「ねじれ補正行列」 $\operatorname{diag}(G_1)^{-1/2}$ の計算}\\
対角線上の各要素の逆平方根をとります：
\begin{align*}
    \operatorname{diag}(G_1)^{-1/2} = \begin{pmatrix} \frac{1}{\sqrt{10000}} & 0 \\ 0 & \frac{1}{\sqrt{1}} \end{pmatrix} = \begin{pmatrix} 0.01 & 0 \\ 0 & 1 \end{pmatrix}
\end{align*}

\textit{ステップ3：変換行列を真の勾配に適用}\\
この補正行列を元の勾配 $g_1$ に掛けます：
\begin{align*}
    \operatorname{diag}(G_1)^{-1/2} g_1 = \begin{pmatrix} 0.01 & 0 \\ 0 & 1 \end{pmatrix} \begin{pmatrix} 100 \\ 1 \end{pmatrix} = \begin{pmatrix} 0.01 \times 100 \\ 1 \times 1 \end{pmatrix} = \begin{pmatrix} 1 \\ 1 \end{pmatrix}
\end{align*}

\textit{ステップ4：最終的な AdaGrad の更新ステップ}\\
更新式に代入します：
\begin{align*}
    \Delta x_{Ada} = -\eta \operatorname{diag}(G_1)^{-1/2} g_1 = -0.1 \begin{pmatrix} 1 \\ 1 \end{pmatrix} = \begin{pmatrix} -0.1 \\ -0.1 \end{pmatrix}
\end{align*}

\paragraph{方向の修正}\ \\
2つの更新ベクトルを比較してみましょう：
\begin{itemize}
    \item \textbf{GD のステップ：} $\begin{pmatrix} -10 \\ -0.1 \end{pmatrix}$ （暴走し、目標から大きく逸れる）
    \item \textbf{AdaGrad のステップ：} $\begin{pmatrix} -0.1 \\ -0.1 \end{pmatrix}$ （安定し、目標を直指する）
\end{itemize}
この2次元平面において、真の勾配 $(100, 1)^T$ は更新方向を $x_1$ 軸へと強く引きずり込みますが、AdaGrad の $G_t^{-1/2}$ 行列が強力な\textbf{「ペナルティと補償」}の役割を果たしました。
具体的には、行列の1つ目の要素 $0.01$ が、$x_1$ 方向の本来100もある狂暴な勾配を100分の1に圧縮して $1$ にし、2つ目の要素 $1$ が、$x_2$ 方向の微弱な勾配 $1$ をそのまま維持しました。

\paragraph{結論}\ \\
行列変換を経た実効的な勾配は $(1, 1)^T$ となりました。これは2次元平面において正確に45度の対角線方向を向いており、渓谷の谷底の中心（最適解）を完璧に指し示しています。この極端な数値例から、$G_t^{-1/2}$ という行列が本質的に「空間の歪み補正器」として機能し、引き伸ばされた等高線地形をアルゴリズムの「目」から見て真円に再スケーリングすることで、毎回最適な方向へ導くメカニズムが理解できます。
\end{proof}
\section{RMSProp}
RMSProp（Root Mean Square Propagation \textbf{二乗平均平方根伝播法}）は、勾配をその大きさの移動平均で割ることによって得られる、適応的学習率を持つ勾配降下法の変種です（Tieleman and Hinton, 2012）。
ステップ $t$ における勾配の要素ごとの2乗の指数移動平均 $v(t)$ は、次のように再帰的に定義されます：
\begin{align*}
    v(t) = \gamma v(t - 1) + (1 - \gamma)g_{t-1}^2
\end{align*}
ここで $\gamma \in (0, 1)$ は指数減衰率を制御する忘却係数です。

\hosoku \textbf{移動平均の一般項の推導}\\
数学的帰納法を用いてこの漸化式を展開すると、以下のようになります：
\begin{align*}
    v(t) &= \gamma (\gamma v(t-2) + (1-\gamma)g_{t-2}^2) + (1-\gamma)g_{t-1}^2 \\
         &= \dots = \gamma^t v(0) + (1 - \gamma)\sum_{i=1}^t \gamma^{t-i}g_i^2
\end{align*}
これらの係数の和は1になります。等比数列の和の公式を用いて証明します：
\begin{align*}
    \gamma^t + (1 - \gamma) \sum_{i=1}^t \gamma^{t-i} 
    &= \gamma^t + (1 - \gamma) \left( \gamma^{t-1} + \gamma^{t-2} + \dots + \gamma^0 \right) \\
    &= \gamma^t + (1 - \gamma) \frac{1 - \gamma^t}{1 - \gamma} = \gamma^t + 1 - \gamma^t = 1
\end{align*}
これにより、$v(t)$ が初期値 $v(0)$（通常は0）と、最近の勾配ほど大きな重みを持たせた過去の勾配の2乗の加重平均であることがわかります。

パラメータの更新規則は以下の通りです：
\begin{align*}
    x(t + 1) = x(t) - \eta \frac{g_t}{\sqrt{|v(t)|}}
\end{align*}
分母の $\sqrt{|v(t)|}$ は勾配の標準偏差の推定値として機能します。

\begin{reidai}{RMSPropによる2次関数の最小化と収束証明}{rmsprop_ex}
実数値関数 $C(x) = \frac{1}{2}x^2$ の最小値を RMSProp メソッドを用いて追跡します。
\end{reidai}
\begin{proof}
この場合の勾配は $g_t = x(t)$ であり、$v(0) = 0$ とすると移動平均は以下のようになります：
\begin{align*}
    v(t) = (1 - \gamma) \sum_{i=1}^t \gamma^{t-i}x(i)^2
\end{align*}
更新式は次のように変形できます：
\begin{align*}
    x(t + 1) = x(t) - \eta \frac{x(t)}{\sqrt{|v(t)|}} = x(t)\left[1 - \eta \frac{1}{\sqrt{|v(t)|}}\right]
\end{align*}
ここで、収縮係数を $\rho_t = 1 - \eta / \sqrt{|v(t)|}$ と定義します。もし $0 < \rho_t < \rho < 1$ を満たす定数 $\rho$ が存在すると仮定すれば、関係式 $x(t+1) = x(t)\rho_t$ と初期条件 $x(0) > 0$ から、数列 $x(t)$ は単調減少であり下に有界（0より大きい）であることがわかります：
\begin{align*}
    0 < x(t) < x(0)\rho^t
\end{align*}
極限をとり、はさみうちの定理（Squeeze Theorem）を用いると、$t \to \infty$ のとき $\rho^t \to 0$ となるため、$x^* = \lim_{t\to\infty} x(t) = 0$ が得られます。

\textbf{二重不等式 $0 < \rho_t < \rho < 1$ の詳細な証明：}

\paragraph{下限 ($0 < \rho_t$) の証明}\ \\
$0 < \rho_t$ は $\dfrac{\eta}{\sqrt{v(t)}} < 1$ と同値です。背理法を用います。もし $x(t)$ が $0$ に収束しないと仮定すると、ある $\epsilon > 0$ が存在し、全ての $t \ge 1$ について $x(t)^2 > \epsilon$ となる。このとき、$v(t)$ を下から評価する：
\begin{align*}
    v(t) &= (1 - \gamma) \sum_{i=1}^t \gamma^{t-i}x(i)^2 > \epsilon(1 - \gamma)\sum_{i=1}^t \gamma^{t-i} \\
         &= \epsilon(1 - \gamma)\dfrac{1 - \gamma^t}{1 - \gamma} = \epsilon(1 - \gamma^t)
\end{align*}
十分大きな $N$ を選び、$t > N$ において $\gamma^t < \dfrac{1}{2}$ となるようにすると、$v(t) > \dfrac{\epsilon}{2}$ となる。
学習率 $\eta$ を $\eta < \dfrac{\epsilon}{2}$ となるように初期設定しておけば、
\begin{align*}
    \dfrac{\eta}{v(t)} < \dfrac{\eta}{\epsilon/2} = \dfrac{2\eta}{\epsilon} < 1 \quad (\forall t > N)
\end{align*}
となり、$0 < \rho_t$ が証明されます。

\paragraph{上限 ($\rho_t < \rho < 1$) の証明}\ \\
$\rho_t < \rho < 1$ は、$v(t) < \left(\frac{\eta}{1-\rho}\right)^2$ と同値であり、数列 $v(t)$ が上に有界であることを意味します。
$x(t)$ は単調減少列であるため、ある定数 $M$ に対して $|x(t)| < M$ （全ての $t$ に対して）が成り立ちます。これを $v(t)$ の定義式に代入して上から評価します：
\begin{align*}
    v(t) &= (1 - \gamma) \sum_{i=1}^t \gamma^{t-i}x(i)^2 < M^2(1 - \gamma)\sum_{i=1}^t \gamma^{t-i} \\
         &= M^2(1 - \gamma^t) < M^2
\end{align*}
これにより $v(t)$ は常に有界となり、対応する $\rho < 1$ を見つけることができます。
\end{proof}

\section{Adam}
Adam（Adaptive Moment Estimation）は、AdaGradとRMSPropから着想を得て、2014年にDiederikとBaによって導入されました。勾配の1次モーメント（平均）と2次モーメント（非中心分散）の指数移動平均を用いて推定し、バイアス補正を適用します。


時間ステップ $t$ での勾配を $g_t = \nabla C(x(t))$ とし、モーメントの初期値を $m(0) = 0$, $v(0) = 0$ とします。減衰率 $\beta_1, \beta_2 \in [0, 1)$ を用いて、次のように更新します：
\begin{align*}
    m(t) &= \beta_1 m(t - 1) + (1 - \beta_1)g_t \quad\cdots \text{SGDのモメンタム項に相当する}\\
    v(t) &= \beta_2 v(t - 1) + (1 - \beta_2)(g_t)^2\quad\cdots \text{RMSPropの移動平均項に相当する}
\end{align*}
これを展開すると、以下の級数表現が得られます：
\begin{align*}
    m(t) = (1 - \beta_1) \sum_{i=1}^t \beta_1^{t-i} g_i, \quad v(t) = (1 - \beta_2) \sum_{i=1}^t \beta_2^{t-i} (g_i)^2
\end{align*}

\hosoku \textbf{期待値とバイアス補正の導出}\\
初期値がゼロであるため、初期のステップでは $m(t)$ や $v(t)$ はゼロに偏ってしまいます（バイアス）。これを補正するために期待値 $\mathbb{E}$ を計算します。勾配が定常的(\textbf{stationarity}\ どの時刻の勾配も、平均的には同じ方向を向いている\ie $\mathbb{E}[g_i] = \mathbb{E}[g_t]$）であると仮定すると：
\begin{align*}
    \mathbb{E}[m(t)] &= \mathbb{E}\left[ (1 - \beta_1) \sum_{i=1}^t \beta_1^{t-i} g_i \right] \\
                     &= (1 - \beta_1) \sum_{i=1}^t \beta_1^{t-i} \mathbb{E}[g_i] \\
                     &= (1 - \beta_1) \mathbb{E}[g_t] \left( \frac{1 - \beta_1^t}{1 - \beta_1} \right) \\
                     &= (1 - \beta_1^t) \mathbb{E}[g_t]
\end{align*}
真の期待値 $\mathbb{E}[g_t]$ に対して $(1 - \beta_1^t)$ 倍のズレが生じていることがわかります。$v(t)$ についても同様に $(1 - \beta_2^t)\mathbb{E}[(g_t)^2]$ となります。
したがって、バイアスを補正したモーメント $\hat{m}(t)$ と $\hat{v}(t)$ は次のように定義されます：
\begin{align*}
    \hat{m}(t) &= m(t) / (1 - \beta_1^t) \\
    \hat{v}(t) &= v(t) / (1 - \beta_2^t)
\end{align*}
最終的なパラメータの更新式は以下の通りです：
\begin{align*}
    x(t + 1) = x(t) - \eta \frac{\hat{m}(t)}{\sqrt{\hat{v}(t)} + \epsilon}
\end{align*}
ここで、$\epsilon > 0$ はゼロ除算を防ぐための小さな定数（デフォルト設定：$\epsilon=10^{-8}$）です。

\section{AdaMax}
AdaMax（Adaptive Maximum method）は、Adamアルゴリズムの無限大ノルム（$L_\infty$ノルム）に基づく変種です。更新規則は以下によって定義されます：
\begin{align*}
    m(t) &= \beta_1 m(t - 1) + (1 - \beta_1)g_t \quad\cdots\text{Adamのものと完全に同一}\\
    u(t) &= \max(\beta_2 u(t - 1), |g(t)|) \\
    x(t) &= x(t - 1) - \frac{\eta}{1 - \beta_1^t} \frac{m(t)}{u(t - 1)}
\end{align*}
モーメントの初期値は $m(0) = u(0) = 0$ とします。
ここで $u(t)$ は、Adamにおける $L_2$ ノルムベースの $v(t)$ の代わりに、過去の勾配の絶対値の最大値を減衰係数 $\beta_2$ を伴って追跡する役割を果たします。この場合、$u(t)$ に対してはバイアス補正を行う必要がありません。

\section{補足証明：行列の等比級数の和と逆行列}\label{app:inverse}

\begin{reidai}{行列の等比級数の和}{geom_matrix}
$S_n = \displaystyle\sum_{k=0}^{n-1} M^k$ と置き、等式の両辺に$I - M$を掛けると、
\[
(\mathbb{I} - M) S_n = (\mathbb{I} - M)(\mathbb{I} + M + \dots + M^{n-1}) = \mathbb{I} - M^n
\]
ここで、$\mathbb{I}$は単位行列(\textbf{Unit Matrix}),\ $M = \begin{pmatrix} 1 & \epsilon \\ -\epsilon a^2 & \mu \end{pmatrix}$\\
行列 $\mathbb{I}-M$ が正則（逆行列が存在）であると仮定すると、
\[ S_n = (\mathbb{I} - M^n)(\mathbb{I} - M)^{-1} \]
となることを示せ
\end{reidai}
\vspace{0.5em}
\begin{proof}
\paragraph{行列の多項式の可換性}\ \\
一般に行列の積は非可換（$AB \neq BA$）であるが、行列 $M$ とその多項式 $P(M)$ は可換である。
$S_n = I + M + \dots + M^{n-1}$ は $M$ の多項式であるため、$M$ および $(I-M)$ と可換になる。
まず、$(\mathbb{I} - M)$ を $S_n$ の\textbf{右側}から掛けた場合を展開する。
\begin{align*}
    S_n (\mathbb{I} - M) &= (\mathbb{I} + M + M^2 + \dots + M^{n-1})(\mathbb{I} - M) \\
    &= (\mathbb{I} + M + \dots + M^{n-1}) - (M + M^2 + \dots + M^n) \\
    &= \mathbb{I} + (M - M) + (M^2 - M^2) + \dots + (M^{n-1} - M^{n-1}) - M^n \\
    &= \mathbb{I} - M^n 
\end{align*}
同様に、左側から掛けても $(\mathbb{I} - M)S_n = \mathbb{I} - M^n$ が成立する。

\paragraph{$M$と$(\mathbb{I} - M)$が可換となる証明}\ \\
行列 $M$ と $(\mathbb{I} - M)$ が可換であることを示す。
\begin{align*}
    M(\mathbb{I} - M) &= M - M^2 \\
    \footnotemark[9](\mathbb{I} - M)M &= M - M^2
\end{align*}
\footnotetext[9]{$\mathbb{I} M = \begin{pmatrix} 1 & 0\\ 0 & 1 \end{pmatrix} \cdot \begin{pmatrix} 1 & \epsilon \\ -\epsilon a^2 & \mu \end{pmatrix} = \begin{pmatrix} 1 & \epsilon \\ -\epsilon a^2 & \mu \end{pmatrix} = \begin{pmatrix} 1 & \epsilon \\ -\epsilon a^2 & \mu \end{pmatrix}\cdot \begin{pmatrix} 1 & 0\\ 0 & 1 \end{pmatrix} = M \mathbb{I} = M$}
したがって、$M(\mathbb{I} - M) = (\mathbb{I} - M)M$ が成立する。
\paragraph{$M$と$(\mathbb{I} - M)^{-1}$が可換となる証明}\ \\
行列 $M$ と $(\mathbb{I} - M)^{-1}$ が可換であることを示す。
\begin{align*}
    M(\mathbb{I} - M)^{-1} &= (\mathbb{I} - M)^{-1}M
\end{align*}
両辺に $(\mathbb{I} - M)$ を掛けると、
\begin{align*}
    M &= (\mathbb{I} - M)^{-1}M(\mathbb{I} - M) \\
    &= (\mathbb{I} - M)^{-1}(M - M^2) \\
    &= (\mathbb{I} - M)^{-1}(\mathbb{I} - M)M \\
    &= M
\end{align*}
したがって、$M(\mathbb{I} - M)^{-1} = (\mathbb{I} - M)^{-1}M$ が成立する。
\paragraph{$(\mathbb{I} - M^n)$と$(\mathbb{I} - M)^{-1}$が可換となる証明}\ \\
行列 $(\mathbb{I} - M^n)$ と $(\mathbb{I} - M)^{-1}$ が可換であることを示す。
\begin{align*}
    (\mathbb{I} - M^n)(\mathbb{I} - M)^{-1} &= \mathbb{I}(\mathbb{I}-M)^{-1} - M^n(\mathbb{I} - M)^{-1} \\
    &= (\mathbb{I} - M)^{-1}\mathbb{I} - (\mathbb{I} - M)^{-1}M^n\\
    &= (\mathbb{I} - M)^{-1}(\mathbb{I} - M^n)
\end{align*}
したがって、$(\mathbb{I} - M^n)(\mathbb{I} - M)^{-1} = (\mathbb{I} - M)^{-1}(\mathbb{I} - M^n)$ が成立する。 \\

\noindent
以上より、行列 $M$、$(\mathbb{I} - M)$、$(\mathbb{I} - M)^{-1}$、$(\mathbb{I} - M^n)$ はすべて互いに可換であることが示された。
したがって、等式の両辺に$(\mathbb{I} - M)^{-1}$を掛ける操作は、左側からでも右側からでも同じ結果をもたらす。
\[ S_n = (\mathbb{I} - M)^{-1}(\mathbb{I} - M^n) = (\mathbb{I} - M^n)(\mathbb{I} - M)^{-1} \] 
となる。
\end{proof}

\section{補足証明：畳み込み級数の収束性解析}

\subsection{命題 5.2 の証明：畳み込み項の極限}\label{app:5.2}

\begin{reidai}{Limit of Convolution Term}{prop_conv_limit}
    実数列 $(a_n)$ が $0$ に収束し、級数 $\sum_{n=0}^\infty b_n$ が絶対収束すると仮定する。このとき、以下の極限が成立する。
    \[ \lim_{n \to \infty} c_n = 0, \quad \text{where } c_n = \sum_{i=0}^{n} a_i b_{n-i} \]
\end{reidai}
\vspace{0.5em}
\begin{proof}\ \\
\noindent
目的は、任意の $\epsilon > 0$ に対して、ある $N$ が存在し、すべての $n > N$ において $|c_n| < \epsilon$ となることを示すことである。
(\ie $\forall\epsilon > 0, \exists N \in \mathbb{N}$ s.t. $\forall \mathbb{N} \ni n > N\implies |c_n| < \epsilon$)\\ 
まず、項の順序を入れ替えて整理する（$j = n-i$ とおく）。
\[
c_n = \sum_{i=0}^{n} a_i b_{n-i} = \sum_{j=0}^{n} b_j a_{n-j}
\]
したがって、以下を使って証明する。
\[
|c_n| = \left| \sum_{j=0}^{n} b_j a_{n-j} \right| \le \sum_{j=0}^{n} |b_j| |a_{n-j}|
\]

\paragraph{定数の準備}
\begin{enumerate}
    \item 仮定より $\sum b_n$ は絶対収束するため、その総和を $\mathbf{B} = \sum_{n=0}^\infty |b_n|$ と置く。$B=0$ ならば自明であるため、$B > 0$ とする。
    \item 仮定より $a_n \to 0$ である。収束する数列は有界であるため、ある定数 $M > 0$ が存在して、すべての $n$ について $|a_n| < M$ が成り立つ。
\end{enumerate}

\paragraph{収束の定義}
\begin{enumerate}
  \item  $\forall \epsilon >0, \sum |b_n|< \infty: \exists N_1 \in \mathbb{N}$ s.t. $\forall k \in \mathbb{N},\ k > N_1\implies \sum_{j=k+1}^{\infty} |b_j|  < \dfrac{\epsilon}{2M}\quad (M>0)$
  \vspace{1em}
  \item  $a_n\to 0\ (n\to \infty)$より、$\forall \epsilon > 0, \exists N_2 \in \mathbb{N}$ s.t. $\forall \mathbb{N}\ni m > N_2\implies |a_m| < \dfrac{\epsilon}{2B}\quad (B>0)$
  
\end{enumerate}

\paragraph{和の分割}\ \\
\noindent
任意の $\epsilon > 0$ を固定する。和を「前半部分」と「後半部分」の2つに分割して計算する。
整数 $K=N_1$を用いて、和を分ける。
\[
|c_n| \le \underbrace{\sum_{j=0}^{K} |b_j| |a_{n-j}|}_{\text{第1項}} + \underbrace{\sum_{j=K+1}^{n} |b_j| |a_{n-j}|}_{\text{第2項}}
\]

\paragraph{第1項の評価（$a$の収束性を使用）}\ \\
\noindent
$n \to \infty$ とすると、各項の $a_{n-j}$ のインデックス $n-j$ は無限大に近づく。
$a_n \to 0$ であるため、すべての $0 \le j \le K$ に対して以下を満たすことができる。
\[
|a_{n-j}| < \frac{\epsilon}{2B}
\]
このとき、$n-j>N_2$ where, $0 \le j \le K$ が成り立つため、$n > N_2 + K$ (ここで、$j = K$)が必要である。そのため、$m = n-j \geq n-K > N_2$ となる。
したがって、第1項は以下のように評価できる。
\[
\text{第1項} = \sum_{j=0}^{K} |b_j| |a_{n-j}| < \frac{\epsilon}{2B} \sum_{j=0}^{K} |b_j| \le \frac{\epsilon}{2B} \cdot B = \frac{\epsilon}{2}
\]

\paragraph{第2項の評価（$b$の絶対収束性を使用）}\ \\
\noindent
$j=K+1 > K=N_1$かつ$\left|a_{n-j}\right| \leq M$であるため、第2項は以下のように評価できる。
\[
\text{第2項} = \sum_{j=K+1}^{n} |b_j| |a_{n-j}| \leq M \sum_{j=K+1}^{n} |b_j| \le M \sum_{j=K+1}^{\infty} |b_j| < M \cdot \frac{\epsilon}{2M} = \frac{\epsilon}{2}
\]

\noindent
以上より、十分大きな $n$ に対して、
\[
|c_n| \le \sum_{j=0}^{n} |b_j| |a_{n-j}| < \frac{\epsilon}{2} + \frac{\epsilon}{2} = \epsilon
\]
が成り立つ。よって、$\displaystyle\lim_{n \to \infty} c_n = 0$ である。
\end{proof}

\subsection{定理 5.3 の証明：畳み込み級数の収束}\label{app:5.3}

\begin{reidai}{Convergence of Convolution Series}{thm_conv_proof}
    級数 $\sum a_n$ が値 $A$ に収束し、級数 $\sum b_n$ が値 $B$ に\textbf{絶対収束}するとする。
    このとき、畳み込み級数 $C_n = \sum_{k=0}^n c_k$ （ただし $c_k = \sum_{i=0}^k a_i b_{k-i}$）は収束し、その和は $AB$ に等しい。
    \[
    \sum_{n=0}^{\infty} c_n = \left(\sum_{n=0}^{\infty} a_n\right) \left(\sum_{n=0}^{\infty} b_n\right)
    \]
\end{reidai}
\vspace{0.5em}
\begin{proof}\ \\
\noindent
各級数の部分和を以下のように定義する。
\[
A_n = \sum_{i=0}^{n} a_i, \quad B_n = \sum_{i=0}^{n} b_i, \quad C_n = \sum_{k=0}^{n} c_k
\]
目的は、$n \to \infty$ のとき $C_n \to AB$ を示すことである。

\paragraph{部分和 $C_n$ の変形}\ \\
\noindent
$C_n$ の定義を展開し、和の順序を入れ替える（下図の三角形領域での和をイメージすると分かりやすい）。
\begin{align*}
    C_n &= \sum_{k=0}^{n} c_k = \sum_{k=0}^{n} \sum_{i=0}^{k} a_i b_{k-i} \\
    &= a_0 b_0 \\
    &+ (a_0 b_1 + a_1 b_0) \\
    &+ \dots \\
    &+ (a_0 b_n + \dots + a_n b_0)
\end{align*}
これを $b_j$ についてまとめ直す（$a$ ではなく $b$ に注目するのは、$\sum b_n$ が絶対収束するという強い条件を持っているためである）。
$b_k$ が係数となる項を集めると、
\[
C_n = \sum_{k=0}^{n} b_k (a_0 + a_1 + \dots + a_{n-k}) = \sum_{k=0}^{n} b_k A_{n-k}
\]
となる。

\paragraph{誤差項への分解}\ \\
\noindent
$A_n \to A$ であるから、誤差項 $\delta_n$ を定義する。
\[
A_n = A + \delta_n \quad (\text{ただし } \lim_{n \to \infty} \delta_n = 0)
\]
これを $C_n$ の式に代入する。
\begin{align*}
    C_n &= \sum_{k=0}^{n} b_k (A + \delta_{n-k}) \\
    &= \sum_{k=0}^{n} b_k A + \sum_{k=0}^{n} b_k \delta_{n-k} \\
    &= A \underbrace{\sum_{k=0}^{n} b_k}_{B_n} + \underbrace{\sum_{k=0}^{n} b_k \delta_{n-k}}_{E_n (\text{誤差項})} \\
    &= A B_n + E_n
\end{align*}

\paragraph{極限の評価}\ \\
\noindent
$n \to \infty$ のときの計算を確認する。
\begin{itemize}
    \item 第1項: $A B_n$ について
    $B_n$ は $B$ に収束するため、$\displaystyle\lim_{n \to \infty} A B_n = AB$ である。
    
    \item 第2項: $E_n = \displaystyle\sum_{k=0}^{n} b_k \delta_{n-k}$ について
    、これは数列 $(b_k)$ と数列 $(\delta_k)$ の畳み込みになっている。
    ここで、以下の条件が満たされている。
    \begin{enumerate}
        \item 数列 $(\delta_n)$ は $0$ に収束する（定義より）。
        \item 級数 $\sum b_n$ は絶対収束する（定理の仮定より）。
    \end{enumerate}
    これはまさに、前述の \textbf{\texttt{Prop.}\ref{tha:prop_conv_limit}} の条件と完全に一致する。
    したがって、命題5.2を適用すると、
    \[
    \lim_{n \to \infty} E_n = 0
    \]
\end{itemize}
\noindent
以上より、
\[
\lim_{n \to \infty} C_n = \lim_{n \to \infty} (A B_n + E_n) = AB + 0 = AB = \left(\sum_{n=0}^{\infty} a_n\right) \left(\sum_{n=0}^{\infty} b_n\right)
\]
となり、定理は証明された。
\end{proof}

\section{補足証明：級数収束の必要条件} \label{app:nec_condition}

\begin{prop}[Necessary Condition for Series Convergence]\ \\
  \noindent
    無限級数 $\displaystyle\sum_{n=0}^{\infty} a_n$ が収束するならば、その一般項 $a_n$ は $0$ に収束する。
    \[
    \sum_{n=0}^{\infty} a_n = S \implies \lim_{n \to \infty} a_n = 0
    \]
\end{prop}
\vspace{0.5em}
\begin{proof}\ \\
\noindent
級数 $\sum a_n$ が値 $S$ に収束するということは、その第 $n$ 部分和 $S_n$ の極限が存在し、それが $S$ であることを意味する。
\[
S_n = \sum_{k=0}^{n} a_k, \quad \text{and} \quad \lim_{n \to \infty} S_n = S
\]
ここで、一般項 $a_n$ （ただし $n \ge 1$）は、隣り合う部分和の差として表現できる。
\[
a_n = S_n - S_{n-1}
\]
この等式の両辺について $n \to \infty$ の極限をとる。
\begin{itemize}
    \item $n \to \infty$ のとき、当然ながら $n-1 \to \infty$ である。
    \item したがって、数列 $(S_n)$ が $S$ に収束するならば、1つ項をずらした数列 $(S_{n-1})$ も同じ値 $S$ に収束する。
    \[ \lim_{n \to \infty} S_{n-1} = S \]
\end{itemize}
極限の線形性（差の極限は極限の差）を用いて計算すると、
\begin{align*}
    \lim_{n \to \infty} a_n &= \lim_{n \to \infty} (S_n - S_{n-1}) \\
    &= \lim_{n \to \infty} S_n - \lim_{n \to \infty} S_{n-1} \\
    &= S - S \\
    &= 0
\end{align*}
よって、$\displaystyle\lim_{n \to \infty} a_n = 0$ が示された。
\end{proof}
\newpage
\begin{rem}[Note on Converse]\ \\
  \noindent
    この命題の逆は真ではないである。
    \ie 「一般項 $a_n \to 0$ であっても、級数 $\sum a_n$ が収束するとは限らない」。\\

    \noindent
    代表的な反例は\textbf{調和級数} $\sum \frac{1}{n}$ である。
    \[ \lim_{n \to \infty} \frac{1}{n} = 0 \quad \text{であるが、} \quad \sum_{n=1}^{\infty} \frac{1}{n} = \infty \quad (\text{発散}) \]
    したがって、$\nabla f(x^n) \to 0$ が確認できたとしても、それだけで $x^n$ が収束する（位置が止まる）保証にはならず、より強い条件（例：Prop 4.4.6 (b) の絶対収束など）が必要となる。
\end{rem}

\section{A Simple Convergence Proof of Adam and Adagrad}\label{pf:adam_and_adagrad}
\subsection{記法}

$d \in \mathbb{N}$ を問題の次元（すなわち最適化する関数のパラメータ数）とし，
$[d] = \{1, 2, \ldots, d\}$ とおく．
関数 $h : \mathbb{R}^d \to \mathbb{R}$ に対して，その勾配を $\nabla h$，勾配の第 $i$ 成分を $\nabla_i h$ と表す．
数値安定性のために，例えば $\epsilon = 10^{-8}$ のような小さな定数 $\epsilon$ を用いる．
列 $(u_n)_{n \in \mathbb{N}}$ が $\forall n \in \mathbb{N},\, u_n \in \mathbb{R}^d$ を満たすとき，
$n \in \mathbb{N}$，$i \in [d]$ に対して $u_{n,i}$ でこの列の第 $n$ 番目の要素の第 $i$ 成分を表す．

最適化したい目的関数を $F : \mathbb{R}^d \to \mathbb{R}$ とする．
$\mathbb{E}[\nabla f(x)] = \nabla F(x)$（$\forall x \in \mathbb{R}^d$）を満たす確率的関数
$f : \mathbb{R}^d \to \mathbb{R}$ が存在し，独立同分布のサンプル列 $(f_n)_{n \in \mathbb{N}^*}$
にアクセス可能なオラクルが存在すると仮定する．
$\mathbb{E}_{n-1}[\,\cdot\,]$ で，$f_1, \ldots, f_{n-1}$ を所与とした条件付き期待値を表す．

機械学習においては，$x$ は線形モデルや深層モデルの重みを表し，
$f$ は個々の学習サンプルやミニバッチから計算される損失を，
$F$ は全体の学習目的関数を表す．
目標は $F$ の臨界点（critical point）を見つけることである．

\subsection{適応的手法}

本稿では，Adagrad~\cite{duchi2011} と Adam~\cite{kingma2015} を統一的な定式化のもとで扱う．
$0 < \beta_2 \leq 1$，$0 \leq \beta_1 < \beta_2$ および非負の列 $(\alpha_n)_{n \in \mathbb{N}^*}$
を仮定する．
出発点 $x_0 \in \mathbb{R}^d$，$m_0 = 0$，$v_0 = 0$ を初期値として，
3つのベクトル $m_n, v_n, x_n \in \mathbb{R}^d$ を次の漸化式で定義する．
すべての反復 $n \in \mathbb{N}^*$ に対して，
\begin{align}
    m_{n,i} &= \beta_1 m_{n-1,i} + \nabla_i f_n(x_{n-1}), \label{eq:update_m} \\
    v_{n,i} &= \beta_2 v_{n-1,i} + \bigl(\nabla_i f_n(x_{n-1})\bigr)^2, \label{eq:update_v} \\
    x_{n,i} &= x_{n-1,i} - \alpha_n \frac{m_{n,i}}{\sqrt{\epsilon + v_{n,i}}}. \label{eq:update_x}
\end{align}

パラメータ $\beta_1$ はHeavy-ball型のモメンタムパラメータ（Polyak, 1964）であり，
$\beta_2$ は各座標における過去の勾配二乗値の指数移動平均の減衰率を制御する．
$\beta_1 = 0$，$\beta_2 = 1$，$\alpha_n = \alpha$（定数）と置くと Adagrad が得られる．
元の Adagrad アルゴリズムは Heavy-ball 型のモメンタムを含んでいないが，
本解析は $\beta_1 > 0$ の場合にも適用可能である．

\paragraph{Adam の補正項について}
元の Adam アルゴリズム（Kingma \& Ba, 2015）では，\eqref{eq:update_m} と
\eqref{eq:update_v} の加重和の代わりに加重平均を用いる．
すなわち，
\begin{equation}
    \tilde{m}_{n,i}
    = (1-\beta_1)\sum_{k=1}^{n} \beta_1^{n-k} \nabla_i f_k(x_{k-1})
    = (1-\beta_1)\, m_{n,i}
\end{equation}
という形になる．
これは，ステップサイズを $\alpha_{\mathrm{adam}} = \alpha \cdot \dfrac{1-\beta_1}{\sqrt{1-\beta_2}}$ 
とおくことと同等である．
さらに元の Adam は，最初の数ステップで $m_n$ と $v_n$ がゼロに偏ることを補正するために，
次の形のステップサイズを用いる：
\begin{equation}
    \alpha_{n,\mathrm{adam}}
    = \alpha \cdot \frac{1-\beta_1}{\sqrt{1-\beta_2}}
    \cdot \underbrace{\frac{1}{1-\beta_1^n}}_{\text{$m_n$ の補正項}}
    \cdot \underbrace{\sqrt{1-\beta_2^n}}_{\text{$v_n$ の補正項}}.
    \label{eq:adam_step}
\end{equation}
各補正項は $1/(1-\beta)$（$\beta \in \{\beta_1, \beta_2\}$）ステップ程度で
その極限値に収束することがわかる．
本稿では，$m_n$ に対する補正項を省略し，$v_n$ に対する補正項のみを残した
次の代替ステップサイズを採用する：
\begin{equation}
    \alpha_n = \alpha(1-\beta_1)\sqrt{\frac{1-\beta_2^n}{1-\beta_2}}.
    \label{eq:step_size}
\end{equation}


\begin{reidai}{$\alpha_{n,\mathrm{adam}}$ の導出}{dd}
元の Adam アルゴリズム（Kingma \& Ba, 2015）では，モメンタム項 $m_n$ と
二乗勾配の指数移動平均 $v_n$ を次の漸化式で定義する：
\begin{align}
    m_{n,i} &= \beta_1 m_{n-1,i} + \nabla_i f_n(x_{n-1}), \label{eq:m_raw}\\
    v_{n,i} &= \beta_2 v_{n-1,i} + \bigl(\nabla_i f_n(x_{n-1})\bigr)^2. \label{eq:v_raw}
\end{align}
ただし，初期値は $m_0 = 0$，$v_0 = 0$ とする．
\end{reidai}
\paragraph{加重和から加重平均への変換}

漸化式~\eqref{eq:m_raw} を展開すると，
\begin{equation}
    m_{n,i}
    = \sum_{k=1}^{n} \beta_1^{n-k} \nabla_i f_k(x_{k-1})
\end{equation}
と表せる．
一方，元の Adam が実際に用いるのは加重\textbf{平均}であり，
正規化係数 $(1-\beta_1)$ を掛けた
\begin{equation}
    \tilde{m}_{n,i}
    = (1-\beta_1)\sum_{k=1}^{n} \beta_1^{n-k} \nabla_i f_k(x_{k-1})
    = (1-\beta_1)\,m_{n,i}
    \label{eq:m_tilde}
\end{equation}
を用いる．同様に $v_n$ についても，
\begin{equation}
    \tilde{v}_{n,i}
    = (1-\beta_2)\sum_{k=1}^{n} \beta_2^{n-k} \bigl(\nabla_i f_k(x_{k-1})\bigr)^2
    = (1-\beta_2)\,v_{n,i}.
    \label{eq:v_tilde}
\end{equation}
この加重平均の採用は，パラメータ更新式
\begin{equation}
    x_{n,i} = x_{n-1,i} - \alpha \cdot \frac{\tilde{m}_{n,i}}{\sqrt{\tilde{v}_{n,i}} + \epsilon}
\end{equation}
において，加重和 $m_{n,i}$ を用いた更新式
\begin{equation}
    x_{n,i} = x_{n-1,i} - \alpha_{\mathrm{adam}} \cdot \frac{m_{n,i}}{\sqrt{v_{n,i}} + \epsilon}
\end{equation}
と等価になることから，グローバルなステップサイズを
\begin{equation}
    \alpha_{\mathrm{adam}}
    = \alpha \cdot \frac{1-\beta_1}{\sqrt{1-\beta_2}}
    \label{eq:alpha_adam_base}
\end{equation}
と置くことと同等である．

\paragraph{初期バイアスの補正項の導入}

$m_0 = 0$，$v_0 = 0$ と初期化しているため，
学習初期（$n$ が小さい時）においては，
$m_n$ および $v_n$ はゼロ方向へのバイアスが生じる．
具体的に，$n$ ステップ後の加重和の正規化係数を計算すると，
\begin{equation}
    \sum_{k=1}^{n} \beta_1^{n-k}(1-\beta_1)
    = 1 - \beta_1^n,
\end{equation}
すなわち $n \to \infty$ の極限で初めて $1$ に収束する．
したがって，有限ステップ $n$ における正規化係数のずれを補正するために，
それぞれ次の\textbf{バイアス補正項}を導入する：
\begin{align}
    \hat{m}_{n,i} &= \frac{\tilde{m}_{n,i}}{1-\beta_1^n}
    = \frac{m_{n,i} (1-\beta_1)}{1-\beta_1^n}, \\
    \hat{v}_{n,i} &= \frac{\tilde{v}_{n,i}}{1-\beta_2^n}
    = \frac{v_{n,i}(1-\beta_2)}{1-\beta_2^n}.
\end{align}
各補正項は，$1/(1-\beta)$ ステップ程度で極限値に収束する
（$\beta \in \{\beta_1, \beta_2\}$）．

\paragraph{補正済みステップサイズの導出}

補正後の推定量 $\hat{m}_{n,i}$，$\hat{v}_{n,i}$ を用いた更新式
\begin{equation}
    x_{n,i}
    = x_{n-1,i}
      - \alpha \cdot \frac{\hat{m}_{n,i}}{\sqrt{\hat{v}_{n,i}} + \epsilon}
    = x_{n-1,i}
      - \alpha \cdot
        \frac{(1-\beta_1)}{1-\beta_1^n} \cdot
        \sqrt{\frac{1-\beta_2^n}{1-\beta_2}} \cdot
        \frac{m_{n,i}}{\sqrt{v_{n,i}} + \epsilon}
\end{equation}
を，加重和 $m_{n,i}$，$v_{n,i}$ をそのまま使った更新式と見比べると，
反復ごとに変化する実効ステップサイズが
\begin{equation}
    \alpha_{n,\mathrm{adam}}
    = \alpha \cdot \frac{1-\beta_1}{\sqrt{1-\beta_2}}
      \cdot \underbrace{\frac{1}{1-\beta_1^n}}_{\substack{m_n \text{ の}\\\text{補正項}}}
      \cdot \underbrace{\sqrt{1-\beta_2^n}}_{\substack{v_n \text{ の}\\\text{補正項}}}
    \label{eq:alpha_full}
\end{equation}
と得られる．これが式~[22] の正体である．
基本ステップサイズ \eqref{eq:alpha_adam_base} に対して，
$m_n$ 向けの補正として $1/(1-\beta_1^n)$ 倍，

\paragraph{簡略化}

式~\eqref{eq:alpha_full} の $m_n$ の補正項 $1/(1-\beta_1^n)$ は，
$v_n$ の補正項と比較して，$1/(1-\beta_1)$ ステップ程度で急速に
極限値 $1$ に収束する（例：$\beta_1=0.9$ では約50ステップ）．
また，$m_n$ の補正項を省略することは学習率ウォームアップと等価であり，
深層学習で広く用いられる手法である．
一方，$v_n$ の補正項を省略すると，$\beta_2 \approx 1$ の場合に
学習初期のステップサイズが過大になり，実際に発散を引き起こすリスクがある．

以上の観察から，本稿では $m_n$ の補正項を省略し，$v_n$ の補正項のみを残した
次の代替ステップサイズを採用する：
\begin{equation}
    \alpha_n
    = \alpha(1-\beta_1)\sqrt{\frac{1-\beta_2^n}{1-\beta_2}}.
    \label{eq:alpha_simplified}
\end{equation}


\subsection{仮定}

以下の3つの仮定を設ける．

\begin{assumption}[目的関数の下有界性]
$F$ は $F^*$ で下に有界である，すなわち，
\begin{equation}
    \forall x \in \mathbb{R}^d,\quad F(x) \geq F^*.
    \label{eq:bounded_below}
\end{equation}
\end{assumption}

\begin{assumption}[確率的勾配の一様有界性]
確率的勾配の $\ell_\infty$ ノルムが一様に概収束で有界である，
すなわち $R \geq \sqrt{\epsilon}$（最終的な上界の簡略化のため $\sqrt{\epsilon}$ を用いる）が存在し，
\begin{equation}
    \forall x \in \mathbb{R}^d,\quad
    \|\nabla f(x)\|_\infty \leq R - \sqrt{\epsilon}
    \quad \text{a.s.}
    \label{eq:grad_bound}
\end{equation}
を満たす．
\end{assumption}

\begin{assumption}[目的関数の滑らかさ]
目的関数が滑らかである，
すなわちその勾配が $\ell_2$ ノルムに関して $L$-Lipschitz 連続である：
\begin{equation}
    \forall x, y \in \mathbb{R}^d,\quad
    \|\nabla F(x) - \nabla F(y)\|_2 \leq L \|x - y\|_2.
    \label{eq:smoothness}
\end{equation}
\end{assumption}
\begin{reidai}{Descent Lemma の証明}{dd}
目的関数 $F : \mathbb{R}^d \to \mathbb{R}$ の勾配が $\ell_2$ ノルムに関して
$L$-Lipschitz 連続であると仮定する，すなわち
\begin{equation}
    \forall\, x, y \in \mathbb{R}^d, \quad
    \|\nabla F(x) - \nabla F(y)\|_2 \leq L\|x - y\|_2.
    \label{eq:L-smooth}
\end{equation}
このとき，任意の $x, y \in \mathbb{R}^d$ に対して次の不等式が成り立つ：
\begin{equation}
    F(y) \leq F(x) + \nabla F(x)^\top (y - x) + \frac{L}{2}\|y - x\|_2^2.
    \label{eq:descent_lemma}
\end{equation}
\end{reidai}

\begin{proof}

\paragraph{微積分学の基本定理による積分表現}\ \\

$F$ は微分可能であるから，微積分学の基本定理を $x$ から $y$ への直線路に適用すると，
\begin{equation}
    F(y) - F(x)
    = \int_0^1 \frac{d}{dt} F\bigl(x + t(y-x)\bigr)\, dt
    = \int_0^1 \nabla F\bigl(x + t(y-x)\bigr)^\top (y - x)\, dt.
    \label{eq:ftoc}
\end{equation}

\paragraph{$\nabla F(x)^\top(y-x)$ の加減算による変形}\ \\

右辺に $\nabla F(x)^\top(y-x)$ を加減することで，
\begin{align}
    F(y) - F(x)
    &= \int_0^1 \nabla F\bigl(x + t(y-x)\bigr)^\top (y-x)\, dt \notag \\
    &= \int_0^1 \nabla F(x)^\top (y-x)\, dt
      + \int_0^1 \Bigl[\nabla F\bigl(x + t(y-x)\bigr) - \nabla F(x)\Bigr]^\top (y-x)\, dt \notag \\
    &= \nabla F(x)^\top (y-x)
      + \int_0^1 \Bigl[\nabla F\bigl(x + t(y-x)\bigr) - \nabla F(x)\Bigr]^\top (y-x)\, dt.
    \label{eq:decomp}
\end{align}
ここで，$\int_0^1 \nabla F(x)^\top(y-x)\, dt = \nabla F(x)^\top(y-x)$ を用いた．

\paragraph{Cauchy--Schwarz不等式を適用}\ \\

式~\eqref{eq:decomp} の第2項を $I$ とおく：
\begin{equation}
    I = \int_0^1 \Bigl[\nabla F\bigl(x + t(y-x)\bigr) - \nabla F(x)\Bigr]^\top (y-x)\, dt.
\end{equation}
Cauchy--Schwarz 不等式 $a^\top b \leq \|a\|_2 \|b\|_2\ (a,b\in \mathbb{R}^d)$ を適用すると，
\begin{equation}
    I \leq \int_0^1
        \bigl\|\nabla F\bigl(x + t(y-x)\bigr) - \nabla F(x)\bigr\|_2
        \cdot \|y - x\|_2\, dt.
    \label{eq:CS}
\end{equation}

\paragraph{Lipschitz 連続性による評価}\ \\

仮定~\eqref{eq:L-smooth} の $L$-Lipschitz 連続性を，
$x$ と $x + t(y-x)$ の対に適用する：
\begin{align}
    \bigl\|\nabla F\bigl(x + t(y-x)\bigr) - \nabla F(x)\bigr\|_2
    &\leq L \bigl\|x + t(y-x) - x\bigr\|_2 \notag \\
    &= L \|t(y-x)\|_2 \notag \\
    &= tL\|y-x\|_2.
    \label{eq:lip_apply}
\end{align}

\paragraph{積分の計算と上界の確定}\ \\

式~\eqref{eq:lip_apply} を式~\eqref{eq:CS} に代入すると，
\begin{align}
    I
    &\leq \int_0^1 tL\|y-x\|_2 \cdot \|y-x\|_2\, dt \notag \\
    &= L\|y-x\|_2^2 \int_0^1 t\, dt \notag \\
    &= L\|y-x\|_2^2 \cdot \frac{1}{2} \notag \\
    &= \frac{L}{2}\|y-x\|_2^2.
    \label{eq:I_bound}
\end{align}

最後、式~\eqref{eq:I_bound} を式~\eqref{eq:decomp} に戻すと，
\begin{equation}
    F(y) - F(x)
    = \nabla F(x)^\top (y-x) + I
    \leq \nabla F(x)^\top (y-x) + \frac{L}{2}\|y-x\|_2^2.
\end{equation}
両辺に $F(x)$ を加えることで，
\begin{equation}
    F(y) \leq F(x) + \nabla F(x)^\top (y-x) + \frac{L}{2}\|y-x\|_2^2
\end{equation}
が得られる．
\end{proof}
\section{主結果（Main Results）と収束性証明}

\subsection{確率的出力点 $\tau$ の構成とその意味}

総反復回数(a number of iterations) $N \in \mathbb{N}^* = \left\{1,2,\cdots\right\}$ を与える．
確率変数 $\tau \in \{0,1,\dots,N-1\}$ を次で定義する：
\begin{equation}
    \forall j < N,\quad
    \mathbb{P}[\tau = j] \propto 1 - \beta_1^{\,N-j}.
    \tag{9}
\end{equation}

すなわち正規化定数
\[
    Z_N := \sum_{k=0}^{N-1} \left(1-\beta_1^{\,N-k}\right)
\]
を用いて
\[
    \mathbb{P}[\tau=j]
    = \frac{1-\beta_1^{\,N-j}}{Z_N}\quad\quad \left(\displaystyle\sum_{j=0}^{N-1} \mathbb{P}[\tau=j] = 1\right)
\]
と書ける．

\paragraph{説明}
\begin{itemize}
    \item $\beta_1=0$ のとき，$\mathbb{P}[\tau=j]$ は一様分布となる
    \item $\mathbb{E}\big[\|\nabla F(x_\tau)\|_2^2\big] = \displaystyle\sum_{j=0}^{N-1} \mathbb{P}[\tau=j] \cdot \mathbb{E}[\|\nabla F(x_j)\|_2^2] $であるため，$\tau$ は $x_n$ の中から無作為に選ばれた出力点を表す
\end{itemize}

\bigskip
\subsection{証明の基本構造}

全ての定理は以下の共通構造に基づく．

\paragraph{Descent Lemma}
ここで、$x_n - x_{n-1} = -\alpha_n u_n\quad (\alpha_n:学習率; u_n:方向更新の式)$ とおくと、Descent Lemma より
\begin{equation}
    F(x_n)
    \leq F(x_{n-1})
    - \alpha_n \nabla F(x_{n-1})^\top u_n
    + \frac{L\alpha_n^2}{2}\|u_n\|_2^2.
\end{equation}

\paragraph{期待値と総和}
両辺の期待値を取り，$n=1,\dots,N$ で総和をとる：
\begin{align}
    \mathbb{E}[F(x_N)]
    &\leq F(x_0)
    - \sum_{n=1}^N \alpha_n
      \mathbb{E}[\nabla F(x_{n-1})^\top u_n]
    + \frac{L}{2}\sum_{n=1}^N \alpha_n^2
      \mathbb{E}[\|u_n\|_2^2].
\end{align}
%%proofの構造はここまでで共通．あとは下降方向の評価を行う．

\begin{proof}\ \\
\noindent
Descent Lemma より，各 $n = 1, \dots, N$ について：
\begin{align}
    F(x_n) \leq F(x_{n-1}) - \alpha_n \nabla F(x_{n-1})^\top u_n + \frac{L\alpha_n^2}{2}\|u_n\|_2^2.
\end{align}
両辺の全期待値 $\mathbb{E}[\,\cdot\,]$ をとると，期待値の単調性より：
\begin{align}
    \mathbb{E}[F(x_n)]
    \leq \mathbb{E}[F(x_{n-1})]
    - \alpha_n \mathbb{E}[\nabla F(x_{n-1})^\top u_n]
    + \frac{L\alpha_n^2}{2} \mathbb{E}[\|u_n\|_2^2].
\end{align}
この不等式は $n = 1, 2, \dots, N$ の各ステップで成立する。

次、上の不等式を $n = 1, \dots, N$ について足し合わせると、
\begin{align}
    \sum_{n=1}^N \mathbb{E}[F(x_n)]
    \leq \sum_{n=1}^N \mathbb{E}[F(x_{n-1})]
    - \sum_{n=1}^N \alpha_n \mathbb{E}[\nabla F(x_{n-1})^\top u_n]
    + \frac{L}{2} \sum_{n=1}^N \alpha_n^2 \mathbb{E}[\|u_n\|_2^2].
\end{align}


左辺から右辺第一項を移項すると：
\begin{align}
    \sum_{n=1}^N \mathbb{E}[F(x_n)] - \sum_{n=1}^N \mathbb{E}[F(x_{n-1})]
    &= \sum_{n=1}^N \Bigl(\mathbb{E}[F(x_n)] - \mathbb{E}[F(x_{n-1})]\Bigr) \notag \\
    &= \bigl(\mathbb{E}[F(x_1)] - \mathbb{E}[F(x_0)]\bigr) \notag \\
    &\quad + \bigl(\mathbb{E}[F(x_2)] - \mathbb{E}[F(x_1)]\bigr) \notag \\
    &\quad + \cdots \notag \\
    &\quad + \bigl(\mathbb{E}[F(x_N)] - \mathbb{E}[F(x_{N-1})]\bigr) \notag \\
    &= \mathbb{E}[F(x_N)] - \mathbb{E}[F(x_0)].
\end{align}
ここで，$x_0$ は確定的な初期点であるから $\mathbb{E}[F(x_0)] = F(x_0)$ が成り立つ。
\noindent
以上によると、
\begin{align}
    \mathbb{E}[F(x_N)]
    &\leq F(x_0)
    - \sum_{n=1}^N \alpha_n
      \mathbb{E}[\nabla F(x_{n-1})^\top u_n]
    + \frac{L}{2}\sum_{n=1}^N \alpha_n^2
      \mathbb{E}[\|u_n\|_2^2]. \tag{52}
\end{align}
\end{proof}
\paragraph{下降方向の評価}
\begin{equation}
    \mathbb{E}[\nabla F(x_{n-1})^\top u_n]
    \geq \frac{1}{2}
    \mathbb{E}\!\left[
        \frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon + v_n}}
    \right]
    - \mathcal{E}_n,
\end{equation}
ただし $\mathcal{E}_n$ は誤差項であり、$u_n$ は更新方向である。

\begin{proof}\ \\
\noindent
更新方向 $u_n$ は Adam/Adagrad の構造より
\begin{align}
    u_n = \frac{\tilde{v}_n}{\sqrt{\epsilon + v_n}},
\end{align}
ここで $\tilde{v}_n$ は $\beta_1$ による指数移動平均（動量付き勾配），
$v_n$ は勾配二乗の累積（二次モーメント推定）である．
よって内積を成分展開すると：
\begin{align}
    \nabla F(x_{n-1})^\top u_n
    = \sum_i \frac{[\nabla F(x_{n-1})]_i \cdot [\tilde{v}_n]_i}{\sqrt{\epsilon + [v_n]_i}}.
\end{align}

\medskip
各成分について：
\begin{align}
    [\tilde{v}_n]_i
    = [\nabla F(x_{n-1})]_i
    + \underbrace{\bigl([\tilde{v}_n]_i - [\nabla F(x_{n-1})]_i\bigr)}_{\text{動量による偏差（ノイズ項）}}.
\end{align}
これを代入すると：
\begin{align}
    \nabla F(x_{n-1})^\top u_n
    &= \sum_i \frac{[\nabla F(x_{n-1})]_i^2}{\sqrt{\epsilon + [v_n]_i}}
     + \sum_i \frac{[\nabla F(x_{n-1})]_i \cdot
       \bigl([\tilde{v}_n]_i - [\nabla F(x_{n-1})]_i\bigr)}
       {\sqrt{\epsilon + [v_n]_i}} \notag \\
    &= \frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon + v_n}}\bigg|_{\text{ベクトル各成分}}
     + \underbrace{\text{（誤差項）}}_{\delta_n}.
\end{align}

\medskip
任意の実数 $a,b$ と正数 $s=1 > 0$ に対して Young の不等式[\ref{pf:youngs_ineq}]より、
\begin{align}
    ab \geq -\frac{s}{2}a^2 - \frac{1}{2s}b^2.
\end{align}
誤差項の各成分に対して $a = [\nabla F(x_{n-1})]_i / (\epsilon + [v_n]_i)^{1/4}$，
$b = ([\tilde{v}_n]_i - [\nabla F(x_{n-1})]_i)/(\epsilon + [v_n]_i)^{1/4}$
として適用すると、
\begin{align}
    \delta_n
    \geq -\frac{1}{2}\frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon+v_n}}
    - C_n
\end{align}
ここで $C_n \geq 0$ は偏差 $\tilde{v}_n - \nabla F(x_{n-1})$ の大きさに依存する残余項である．

\medskip
したがって、
\begin{align}
    \nabla F(x_{n-1})^\top u_n
    \geq \frac{1}{2}\frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon+v_n}}
    - C_n.
\end{align}
両辺の期待値をとり，$\mathcal{E}_n := \mathbb{E}[C_n] \geq 0$ と定義すれば：
\begin{align}
    \mathbb{E}[\nabla F(x_{n-1})^\top u_n]
    \geq \frac{1}{2}
    \mathbb{E}\!\left[
        \frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon + v_n}}
    \right]
    - \mathcal{E}_n.
    \tag{57}
\end{align}
ここで $\mathcal{E}_n$ は動量パラメータ $\beta_1$ による偏差（$\tilde{v}_n \neq \nabla F(x_{n-1})$）
から生じる誤差項である。
\end{proof}

\paragraph{望遠鏡和}
$F(x_N)\geq F^*$ より
\begin{equation}
    \sum_{n=1}^N \alpha_n
    \mathbb{E}\!\left[
        \frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon + v_n}}
    \right]
    \leq
    2(F(x_0)-F^*) + 2\sum_{n=1}^N \alpha_n \mathcal{E}_n
    + L\sum_{n=1}^N \alpha_n^2 \mathbb{E}[\|u_n\|^2].
\end{equation}
\begin{proof}\ \\
\noindent
下降方向の評価より、各$n$ について：
\begin{align}
    \mathbb{E}[\nabla F(x_{n-1})^\top u_n]
    \geq \frac{1}{2}\mathbb{E}\!\left[
        \frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon + v_n}}
    \right] - \mathcal{E}_n.
\end{align}
下界で置き換えると右辺は大きくなるので、
\begin{align}
    \mathbb{E}[F(x_N)]
    &\leq F(x_0)
    - \sum_{n=1}^N \alpha_n
      \left(
        \frac{1}{2}\mathbb{E}\!\left[
          \frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon+v_n}}
        \right]
        - \mathcal{E}_n
      \right)
    + \frac{L}{2}\sum_{n=1}^N \alpha_n^2 \mathbb{E}[\|u_n\|^2].
\end{align}
括弧を展開すると：
\begin{align}
    \mathbb{E}[F(x_N)]
    &\leq F(x_0)
    - \frac{1}{2}\sum_{n=1}^N \alpha_n
      \mathbb{E}\!\left[
        \frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon+v_n}}
      \right]
    + \sum_{n=1}^N \alpha_n \mathcal{E}_n
    + \frac{L}{2}\sum_{n=1}^N \alpha_n^2 \mathbb{E}[\|u_n\|^2].
    \label{eq:after_sub}
\end{align}

\medskip
$F^*$ は $F$ の大域的最小値であるから，任意の $x$ に対して $F(x) \geq F^*$．
特に $x = x_N$ として期待値をとれば：
\begin{align}
    F^* \leq \mathbb{E}[F(x_N)].
\end{align}
これを式~\eqref{eq:after_sub} と合わせると：
\begin{align}
    F^*
    \leq \mathbb{E}[F(x_N)]
    \leq F(x_0)
    - \frac{1}{2}\sum_{n=1}^N \alpha_n
      \mathbb{E}\!\left[
        \frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon+v_n}}
      \right]
    + \sum_{n=1}^N \alpha_n \mathcal{E}_n
    + \frac{L}{2}\sum_{n=1}^N \alpha_n^2 \mathbb{E}[\|u_n\|^2].
\end{align}

\medskip
目標項を左辺に移項すると：
\begin{align}
    \frac{1}{2}\sum_{n=1}^N \alpha_n
    \mathbb{E}\!\left[
      \frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon+v_n}}
    \right]
    \leq
    F(x_0) - F^*
    + \sum_{n=1}^N \alpha_n \mathcal{E}_n
    + \frac{L}{2}\sum_{n=1}^N \alpha_n^2 \mathbb{E}[\|u_n\|^2].
\end{align}
両辺を $2$ 倍すれば：
\begin{align}
    \sum_{n=1}^N \alpha_n
    \mathbb{E}\!\left[
        \frac{\|\nabla F(x_{n-1})\|_2^2}{\sqrt{\epsilon + v_n}}
    \right]
    \leq
    2(F(x_0)-F^*) + 2\sum_{n=1}^N \alpha_n \mathcal{E}_n
    + L\sum_{n=1}^N \alpha_n^2 \mathbb{E}[\|u_n\|^2].
    \tag{65}
\end{align}
\end{proof}

\paragraph{$\tau$ の導入}
定義より
\begin{align}
    \mathbb{E}[\|\nabla F(x_\tau)\|^2]
    &= \sum_{j=0}^{N-1}
    \frac{1-\beta_1^{\,N-j}}{Z_N}
    \mathbb{E}[\|\nabla F(x_j)\|^2].
\end{align}
\paragraph{$\sum_{n=1}^N \frac{g_n^2}{\sqrt{\epsilon + v_n}}$ の評価}\ \\
\noindent
各座標 $i$ を固定する（以下，座標添字 $i$ を省略）．
$v_n = \sum_{k=1}^n g_k^2$（$v_0 = 0$）とするとき，
\[
    \sum_{n=1}^N \frac{g_n^2}{\sqrt{\epsilon + v_n}}
    \leq 2\sqrt{v_N}.
\]

\begin{proof}\ \\
\noindent
$v_n$ の定義 $v_n = \sum_{k=1}^n g_k^2$ より，
\[
    v_n - v_{n-1} = g_n^2, \qquad n = 1, 2, \ldots, N.
    \tag{L1}
\]
これにより，左辺の各項を
\[
    \frac{g_n^2}{\sqrt{\epsilon + v_n}}
    = \frac{v_n - v_{n-1}}{\sqrt{\epsilon + v_n}}
\]
と書き換えることができる．

\medskip
\noindent
関数 $f(t) = \sqrt{\epsilon + t}$（$t \geq 0$）を考える．
一階・二階導関数は
\[
    f'(t) = \frac{1}{2\sqrt{\epsilon + t}} > 0, \qquad
    f''(t) = -\frac{1}{4(\epsilon + t)^{3/2}} < 0
\]
であるから，$f$ は狭義凹関数（strictly concave）である．

凹関数の基本性質として，任意の $a \leq b$ に対して
\[
    f(b) - f(a) \geq f'(b)\,(b - a)
    \tag{L2}
\]
が成立する（凹関数では接線が曲線の\emph{上方}にあるため，
右端点 $b$ での傾き $f'(b)$ は区間 $[a,b]$ 上の平均変化率
$\frac{f(b)-f(a)}{b-a}$ \emph{以下}である．
したがって $f'(b)(b-a) \leq f(b)-f(a)$ が成り立つ）．

\medskip
\noindent
$a = v_{n-1}$，$b = v_n$（$g_n^2 \geq 0$ より $v_n \geq v_{n-1}$）を代入すると，
\[
    \sqrt{\epsilon + v_n} - \sqrt{\epsilon + v_{n-1}}
    \geq \frac{1}{2\sqrt{\epsilon + v_n}}\,(v_n - v_{n-1})
    = \frac{g_n^2}{2\sqrt{\epsilon + v_n}}.
\]
両辺を $2$ 倍して移項すると，
\[
    \frac{g_n^2}{\sqrt{\epsilon + v_n}}
    \leq 2\!\left(\sqrt{\epsilon + v_n} - \sqrt{\epsilon + v_{n-1}}\right).
    \tag{L3}
\]

\medskip
\noindent
(L3) を $n = 1, \ldots, N$ について加えると，
\[
    \sum_{n=1}^N \frac{g_n^2}{\sqrt{\epsilon + v_n}}
    \leq 2\sum_{n=1}^N
      \!\left(\sqrt{\epsilon + v_n} - \sqrt{\epsilon + v_{n-1}}\right).
\]
右辺は Telescoping Sum（望遠鏡和）であり，中間項がすべて相殺されて
\begin{align}
    2\sum_{n=1}^N \!\left(\sqrt{\epsilon + v_n} - \sqrt{\epsilon + v_{n-1}}\right)
    &= 2\!\left(\sqrt{\epsilon + v_N} - \sqrt{\epsilon + v_0}\right) \notag\\
    &= 2\!\left(\sqrt{\epsilon + v_N} - \sqrt{\epsilon}\right),
    \tag{L4}
\end{align}
ここで最後の等号には $v_0 = 0$ を用いた．

\medskip
\noindent
$\sqrt{\epsilon} \geq 0$ より，
\[
    \sqrt{\epsilon + v_N} - \sqrt{\epsilon} \leq \sqrt{\epsilon + v_N}.
    \tag{L5}
\]
さらに $\epsilon \geq 0$ より $\sqrt{\epsilon + v_N} \leq \sqrt{\epsilon} + \sqrt{v_N}$
が成り立つが，ここではより直接的に $\epsilon$ を捨てて
\[
    \sqrt{\epsilon + v_N} \leq \sqrt{\epsilon} + \sqrt{v_N}
\]
を用いることもできる．
ここでは $\epsilon$ が十分小さい設定で量級に影響しないため，
\[
    \sqrt{\epsilon + v_N} - \sqrt{\epsilon}
    \leq \sqrt{\epsilon + v_N}
    \leq \sqrt{\epsilon} + \sqrt{v_N}
\]
の最右辺からさらに $\sqrt{\epsilon}$ を落として $\sqrt{v_N}$ とする代わりに，
(L5) から直接
\[
    2\!\left(\sqrt{\epsilon + v_N} - \sqrt{\epsilon}\right)
    \leq 2\sqrt{\epsilon + v_N}
\]
と評価し，最終的に $\epsilon$ を無視して
\[
    \leq 2\sqrt{v_N}
\]
とまとめる
（厳密には $2\sqrt{\epsilon + v_N}$ が正確な上界であるが，
$\epsilon > 0$ は正則化パラメータとして固定されており，
後続の評価 $v_{N} \leq NR^2$ を代入する際に $\epsilon$ 項は
主要項に吸収されるため，量級に影響しない）．

\medskip
\noindent
(L3)–(L5) によると
\[
    \sum_{n=1}^N \frac{g_n^2}{\sqrt{\epsilon + v_n}}
    \leq 2\!\left(\sqrt{\epsilon + v_N} - \sqrt{\epsilon}\right)
    \leq 2\sqrt{\epsilon + v_N}
    \leq 2\sqrt{v_N} + 2\sqrt{\epsilon}.
\]
論文の評価では $\epsilon$ 項を無視し（あるいは $\epsilon \to 0$ の極限で），
\[
    \displaystyle\sum_{n=1}^N \frac{g_{n,i}^2}{\sqrt{\epsilon + v_{n,i}}}
    \leq 2\sqrt{v_{N,i}}
\]
が得られる
\end{proof}


\bigskip
\subsection{定理1：Adagrad（モーメンタムなし）}

\begin{thm}
$\beta_1=0,\ \beta_2=1,\ \alpha_n=\alpha$ のとき，
\begin{equation}
    \mathbb{E}[\|\nabla F(x_\tau)\|^2]
    \leq
    \frac{2R(F(x_0)-F^*)}{\alpha\sqrt{N}}
    + \frac{1}{\sqrt{N}}(4dR^2 + \alpha dRL)\ln\!\left(1+\frac{NR^2}{\epsilon}\right).
\end{equation}
\end{thm}

\begin{proof}\ \\
\noindent
\textbf{目標収束率の導出}\\
\noindent
目標は以下の収束率を示すことである：
\[
    \mathbb{E}\!\left[\|\nabla F(x_\tau)\|^2\right]
    \leq 2R\,\frac{F(x_0)-F_*}{\alpha\sqrt{N}}
    + \frac{1}{\sqrt{N}}\left(4dR^2 + \alpha dRL\right)
    \ln\!\left(1+\frac{NR^2}{\epsilon}\right).
\]

\noindent
\textbf{Descent Lemma による関数値の降下}\\
\noindent
$F$ が $L$-滑らかであることより，各ステップで以下が成立する：
\[
    F(x_{n+1}) \leq F(x_n) - \alpha\langle \nabla F(x_n),\, u_n \rangle
    + \frac{\alpha^2 L}{2}\|u_n\|^2.
\]
移項すると，
\[
    \alpha\langle \nabla F(x_n), u_n \rangle
    \leq F(x_n) - F(x_{n+1}) + \frac{\alpha^2 L}{2}\|u_n\|^2.
\]
$n = 1, \ldots, N$ について総和をとると，右辺は Telescoping により，
\[
    \alpha \sum_{n=1}^N \langle \nabla F(x_n), u_n \rangle
    \leq F(x_0) - F(x_{N+1}) + \frac{\alpha^2 L}{2}\sum_{n=1}^N \|u_n\|^2.
\]
$F(x_{N+1}) \geq F_*$ より $F(x_0) - F(x_{N+1}) \leq F(x_0) - F_*$ なので，
両辺を $\alpha$ で割ると，
\[
    \sum_{n=1}^N \langle \nabla F(x_n), u_n \rangle
    \leq \frac{F(x_0)-F_*}{\alpha}
    + \frac{\alpha L}{2}\sum_{n=1}^N \|u_n\|^2.
    \tag{A}
\]

\noindent
\textbf{$\langle \nabla F(x_n), u_n \rangle$ と $\|\nabla F\|^2$ の関係}\\
\noindent
AdaGrad の更新量 $u_n$ の第 $i$ 成分は
\[
    u_{n,i} = \frac{g_{n,i}}{\sqrt{\epsilon + v_{n,i}}}
\]
であるから，
\[
    \langle \nabla F(x_n), u_n \rangle
    = \sum_{i=1}^d \frac{\nabla_i F(x_n) \cdot g_{n,i}}{\sqrt{\epsilon+v_{n,i}}}.
\]
確率的勾配の不偏性 $\mathbb{E}[g_{n,i}] = \nabla_i F(x_n)$ より，期待値をとると，
\[
    \mathbb{E}\!\left[\langle \nabla F(x_n), u_n \rangle\right]
    = \sum_{i=1}^d \frac{(\nabla_i F(x_n))^2}{\sqrt{\epsilon+v_{n,i}}}.
\]
仮定 $\|g_n\|_\infty \leq R$ より各座標で $\sqrt{\epsilon + v_{n,i}} \leq \sqrt{v_{N,i}} \leq \sqrt{NR^2} = R\sqrt{N}$ であるが，
より単純な評価として $\sqrt{\epsilon + v_{n,i}} \leq 2R$ を用いると，
\[
    \sum_{i=1}^d \frac{(\nabla_i F(x_n))^2}{\sqrt{\epsilon+v_{n,i}}}
    \geq \frac{1}{2R}\sum_{i=1}^d (\nabla_i F(x_n))^2
    = \frac{\|\nabla F(x_n)\|^2}{2R}.
\]
すなわち，
\[
    \mathbb{E}\!\left[\langle \nabla F(x_n), u_n \rangle\right]
    \geq \frac{\|\nabla F(x_n)\|^2}{2R}.
    \tag{B}
\]

\noindent
$u_{n,i}$ の定義より，
\[
    \|u_n\|^2 = \sum_{i=1}^d \frac{g_{n,i}^2}{\epsilon + v_{n,i}}.
\]
$n = 1, \ldots, N$ について総和をとり，和の順序を交換すると，
\[
    \sum_{n=1}^N \|u_n\|^2
    = \sum_{i=1}^d \sum_{n=1}^N \frac{g_{n,i}^2}{\epsilon + v_{n,i}}.
\]
$g_{n,i}^2 = v_{n,i} - v_{n-1,i}$ および $f(t)=1/t$ の単調減少性より，
各項は右端点矩形として積分で上から抑えられる：
\[
    \frac{v_{n,i}-v_{n-1,i}}{\epsilon+v_{n,i}}
    \leq \int_{v_{n-1,i}+\epsilon}^{v_{n,i}+\epsilon} \frac{dt}{t}.
\]
$n$ について総和をとると右辺が Telescoping し，
\[
    \sum_{n=1}^N \frac{g_{n,i}^2}{\epsilon+v_{n,i}}
    \leq \int_\epsilon^{v_{N,i}+\epsilon}\frac{dt}{t}
    = \ln\!\left(1+\frac{v_{N,i}}{\epsilon}\right).
\]
さらに $v_{N,i} \leq NR^2$ を代入し，$d$ 座標について総和をとると，
\[
    \sum_{n=1}^N \|u_n\|^2
    \leq \sum_{i=1}^d \ln\!\left(1+\frac{v_{N,i}}{\epsilon}\right)
    \leq \sum_{i=1}^d \ln\!\left(1+\frac{NR^2}{\epsilon}\right)
    = d\ln\!\left(1+\frac{NR^2}{\epsilon}\right).
    \tag{C}
\]
\textbf{(B),(C)を(A)に代入すると}\\
\noindent
$v_{n,i} = \sum_{k=1}^n g_{k,i}^2$ には現在のステップの $g_{n,i}$ が含まれるため，
$g_{n,i}$ と $v_{n,i}$ は独立でない．
このため，期待値を展開すると主項の他に誤差項が生じ，
(B) で使った下界を用いると次の式が得られる：
\[
    \frac{1}{2R}\sum_{n=1}^N \mathbb{E}\!\left[\|\nabla F(x_n)\|^2\right]
    \leq \frac{F(x_0)-F_*}{\alpha}
    + \frac{\alpha L}{2}\cdot d\ln\!\left(1+\frac{NR^2}{\epsilon}\right)
    + \underbrace{\sum_{i=1}^d\sum_{n=1}^N \frac{g_{n,i}^2}{\sqrt{\epsilon+v_{n,i}}}}_{\text{交差誤差項}}.
\]
両辺に $2R$ を掛けると，
\[
    \sum_{n=1}^N \mathbb{E}\!\left[\|\nabla F(x_n)\|^2\right]
    \leq \frac{2R(F(x_0)-F_*)}{\alpha}
    + \alpha dRL\ln\!\left(1+\frac{NR^2}{\epsilon}\right)
    + 2R\sum_{i=1}^d\sum_{n=1}^N \frac{g_{n,i}^2}{\sqrt{\epsilon+v_{n,i}}}.
    \tag{D}
\]

\noindent
各座標 $i$ ごとに
\[
    \sum_{n=1}^N \frac{g_{n,i}^2}{\sqrt{\epsilon+v_{n,i}}}
    \leq 2\sqrt{v_{N,i}} \leq 2\sqrt{NR^2} = 2R\sqrt{N}
\]
が成立するから，全座標 $i=1,\ldots,d$ について和をとると，
\[
    \sum_{i=1}^d\sum_{n=1}^N \frac{g_{n,i}^2}{\sqrt{\epsilon+v_{n,i}}}
    \leq 2dR\sqrt{N}.
\]
これを (D) の誤差項に代入し，$2R \cdot 2dR\sqrt{N} = 4dR^2\sqrt{N}$ よると，
\[
    \sum_{n=1}^N \mathbb{E}\!\left[\|\nabla F(x_n)\|^2\right]
    \leq \frac{2R(F(x_0)-F_*)}{\alpha}
    + \alpha dRL\ln\!\left(1+\frac{NR^2}{\epsilon}\right)
    + 4dR^2\sqrt{N}.
    \tag{D'}
\]

\noindent
$\tau$ は $\{1,\ldots,N\}$ 上の一様ランダムな添字であるから，
\[
    \mathbb{E}\!\left[\|\nabla F(x_\tau)\|^2\right]
    = \frac{1}{N}\sum_{n=1}^N \mathbb{E}\!\left[\|\nabla F(x_n)\|^2\right].
\]
(D') の両辺を $N$ で割ると，精確な上界として
\begin{equation}
    \mathbb{E}\!\left[\|\nabla F(x_\tau)\|^2\right]
    \leq \frac{2R(F(x_0)-F_*)}{\alpha N}
    + \frac{\alpha dRL}{N}\ln\!\left(1+\frac{NR^2}{\epsilon}\right)
    + \frac{4dR^2\sqrt{N}}{N}. \label{eq:exact}
\end{equation}
$\dfrac{\sqrt{N}}{N} = \dfrac{1}{\sqrt{N}}$ より第三項は $\dfrac{4dR^2}{\sqrt{N}}$ である．

\medskip
\noindent
\underline{（第一項）：$\dfrac{1}{N} \leq \dfrac{1}{\sqrt{N}}$}

$N \geq 1$ のとき $\sqrt{N} \leq N$ であるから $\dfrac{1}{N} \leq \dfrac{1}{\sqrt{N}}$ が成立する．
これを第一項に適用すると，
\[
    \frac{2R(F(x_0)-F_*)}{\alpha N}
    \;\leq\;
    \frac{2R(F(x_0)-F_*)}{\alpha\sqrt{N}}.
\]

\medskip
\noindent
\underline{ （第二項）：$\dfrac{1}{\sqrt{N}} \leq 1$}

第二項を分解すると，
\[
    \frac{\alpha dRL}{N}\ln\!\left(1+\frac{NR^2}{\epsilon}\right)
    = \frac{1}{\sqrt{N}} \cdot \frac{\alpha dRL}{\sqrt{N}} \cdot
      \ln\!\left(1+\frac{NR^2}{\epsilon}\right).
\]
ここで $N \geq 1$ のとき $\dfrac{1}{\sqrt{N}} \leq 1$ が成立するから，
内側の $\dfrac{1}{\sqrt{N}}$ を $1$ で上から抑えると，
\[
    \frac{1}{\sqrt{N}} \cdot \frac{\alpha dRL}{\sqrt{N}} \cdot
      \ln\!\left(1+\frac{NR^2}{\epsilon}\right)
    \;\leq\;
    \frac{1}{\sqrt{N}} \cdot \alpha dRL \cdot
      \ln\!\left(1+\frac{NR^2}{\epsilon}\right).
\]

\medskip
\noindent
\underline{ （第三項）：追加のなし}

第三項 $\dfrac{4dR^2}{\sqrt{N}}$ はすでに $\dfrac{1}{\sqrt{N}}$ の形であるから，
そのまま
\[
    \frac{4dR^2}{\sqrt{N}}
    = \frac{1}{\sqrt{N}} \cdot 4dR^2 \cdot
      \ln\!\left(1+\frac{NR^2}{\epsilon}\right)^0.
\]
ただし，第二項と括弧にまとめるために，
$\ln\!\left(1+NR^2/\epsilon\right) \geq 1$（$N \geq 1$, $R > 0$, $\epsilon > 0$ のとき）
を用いて $4dR^2 \leq 4dR^2 \cdot \ln\!\left(1+NR^2/\epsilon\right)$
と上から抑え，第二項と同じ $\ln$ 因子を持たせる：
\[
    \frac{4dR^2}{\sqrt{N}}
    \;\leq\;
    \frac{1}{\sqrt{N}} \cdot 4dR^2 \cdot
      \ln\!\left(1+\frac{NR^2}{\epsilon}\right).
\]

\medskip
\noindent
したがって、\\
\noindent
以上三段階のを式~\eqref{eq:exact} に適用すると，
\[
    \mathbb{E}\!\left[\|\nabla F(x_\tau)\|^2\right]
    \;\leq\;
    \frac{2R(F(x_0)-F_*)}{\alpha\sqrt{N}}
    + \frac{1}{\sqrt{N}} \cdot \alpha dRL \cdot \ln\!\left(1+\frac{NR^2}{\epsilon}\right)
    + \frac{1}{\sqrt{N}} \cdot 4dR^2 \cdot \ln\!\left(1+\frac{NR^2}{\epsilon}\right).
\]
第二項と第三項を $\dfrac{1}{\sqrt{N}}\ln\!\left(1+NR^2/\epsilon\right)$ でくくることで，
\[
    \mathbb{E}\!\left[\|\nabla F(x_\tau)\|^2\right]
    \leq 2R\,\frac{F(x_0)-F_*}{\alpha\sqrt{N}}
    + \frac{1}{\sqrt{N}}\!\left(4dR^2 + \alpha dRL\right)
    \ln\!\left(1+\frac{NR^2}{\epsilon}\right).
    \tag{10}
\]
右辺は $N\to\infty$ のとき $\dfrac{\ln N}{\sqrt{N}}\to 0$ の速度で減少し，
アルゴリズムが勾配ゼロの点に収束することが保証される．
\end{proof}

%     F(x_n) \leq F(x_{n-1})

\begin{lemma}[A.1 (モメンタム付き適応的更新は下降方向にほぼ沿う)]
\label{lemma:A1}
$x_0 \in \mathbb{R}^d$ を出発点とし，系 (A.1) で定義されるイテレートを
非減少な $(\alpha_j)_{j \in \mathbb{N}^*}$ のもとで考える．
条件 (6), (7), (8) および $0 \leq \beta_1 < \beta_2 \leq 1$ を仮定する．
このとき，任意のイテレーション $n \in \mathbb{N}^*$ に対して，
\begin{equation}
    \begin{aligned}
        \mathbb{E}\!\left[\sum_{i \in [d]} G_{n,i} \frac{m_{n,i}}{\sqrt{\epsilon + v_{n,i}}}\right] &\geq\frac{1}{2} \sum_{i \in [d]} \sum_{k=0}^{n-1} \beta_1^k
        \mathbb{E}\!\left[\frac{G_{n-k,i}^2}{\sqrt{\epsilon + \tilde{v}_{n,k+1,i}}}\right] \\
        &- \frac{\alpha_n^2 L^2}{4R\sqrt{1-\beta_1}}
          \sum_{l=1}^{n-1} \|u_{n-l}\|_2^2 \sum_{k=l}^{n-1} \frac{\beta_1^k}{\sqrt{k}} \\
        &- \frac{3R}{\sqrt{1-\beta_1}}
          \sum_{k=0}^{n-1} \left(\frac{\beta_1}{\beta_2}\right)^k \sqrt{k+1}\,\|U_{n-k}\|_2^2.
    \end{aligned}
    \label{eq:lemmaA1}
\end{equation}
\end{lemma}

\begin{proof}
証明中は，式~(A.13)（ヤング不等式\ref{pf:youngs_ineq}）を繰り返し用いる：
\begin{equation}
    \forall\,\lambda > 0,\; x, y \in \mathbb{R},\quad
    xy \leq \frac{\lambda}{2} x^2 + \frac{y^2}{2\lambda}.
    \tag{A.13}
    \label{eq:young}
\end{equation}

\medskip
\noindent
\textbf{モメンタム項の展開と分解}

\smallskip
\noindent
以下では $G_n = \nabla F(x_{n-1})$（真の勾配），
$g_n = \nabla f_n(x_{n-1})$（ランダム勾配），
$u_{n,i} = m_{n,i}/\sqrt{\epsilon+v_{n,i}}$（モメンタム付き更新），
$U_{n,i} = g_{n,i}/\sqrt{\epsilon+v_{n,i}}$（モメンタムなし更新）
を用いる（式 (A.5), (A.6)）．

モメンタムの漸化式 $m_{n,i} = \beta_1 m_{n-1,i} + g_{n,i}$ を展開すると，
\begin{equation}
    m_{n,i} = \sum_{k=0}^{n-1} \beta_1^k\, g_{n-k,i}.
\end{equation}
これを用いて左辺の内積を展開する：
\begin{equation}
    \sum_{i \in [d]} G_{n,i} \frac{m_{n,i}}{\sqrt{\epsilon + v_{n,i}}}
    = \sum_{i \in [d]} \sum_{k=0}^{n-1} \beta_1^k\, G_{n,i}\,
      \frac{g_{n-k,i}}{\sqrt{\epsilon + v_{n,i}}}.
    \label{eq:expand}
\end{equation}
ここで，$G_{n,i}$（現在時刻 $n$ における期待勾配の第 $i$ 成分）を，
$G_{n-k,i}$（過去時刻 $n-k$ における期待勾配の第 $i$ 成分）に置き換えるため，
加減算を行う：
\begin{equation}
    \sum_{i \in [d]} \sum_{k=0}^{n-1} \beta_1^k\, G_{n,i}\,
    \frac{g_{n-k,i}}{\sqrt{\epsilon + v_{n,i}}}
    = \underbrace{\sum_{i \in [d]} \sum_{k=0}^{n-1} \beta_1^k\, G_{n-k,i}\,
      \frac{g_{n-k,i}}{\sqrt{\epsilon + v_{n,i}}}}_{\displaystyle A}
    + \underbrace{\sum_{i \in [d]} \sum_{k=0}^{n-1} \beta_1^k\,
      (G_{n,i} - G_{n-k,i})\, \frac{g_{n-k,i}}{\sqrt{\epsilon + v_{n,i}}}}_{\displaystyle B}.
    \label{eq:AB_decomp}
    \tag{A.14}
\end{equation}

\noindent\textit{直感：}
$G_{n,i}$（現在の期待勾配）と $g_{n-k,i}$（過去の確率的勾配）は，
異なる時刻のものであるため相関を持ちうる．
そこで $G_{n,i}$ を $G_{n-k,i}$（同時刻の期待勾配）に置き換えることで，
$A$ 項では $G_{n-k,i}$ と $g_{n-k,i}$ が同時刻となり，
Lemma~5.1 と同様の議論が適用可能になる．
$B$ 項はその置き換えによる誤差項であり，$F$ の $L$-滑らかさで制御する．

\bigskip
\noindent
\textbf{B項の上界評価（$L$-滑らかさとヤング不等式）}

\smallskip
\noindent
ここでは $i \in [d]$ および $0 \leq k \leq n-1$ を固定する．
以下では省略記法として
$G = G_{n-k,i}$，$g = g_{n-k,i}$，$v = v_{n,i}$ を用いる．

\eqref{eq:young}（ヤング不等式\ref{pf:youngs_ineq}）を，
\begin{equation}
    \lambda = \frac{\sqrt{1-\beta_1}}{\color{red} 2R\sqrt{k+1}},\quad
    x = |G_{n,i} - G_{n-k,i}|,\quad
    y = \frac{|g_{n-k,i}|}{\sqrt{\epsilon + v_{n,i}}}
    \label{eq:lambda_B}
\end{equation}
と選んで適用すると，
\begin{equation}
    |B| \leq \sum_{i \in [d]} \sum_{k=0}^{n-1} \beta_1^k
    \left(
      \frac{\sqrt{1-\beta_1}}{4R}\sqrt{k+1}\,(G_{n,i} - G_{n-k,i})^2
      + \frac{R\sqrt{k+1}}{\sqrt{1-\beta_1}} \frac{g_{n-k,i}^2}{\epsilon + v_{n,i}}
    \right).
    \tag{A.15}
    \label{eq:B_bound}
\end{equation}

次に，$\epsilon + v_{n,i}$ と $\epsilon + v_{n-k,i}$ の関係を評価する．
$v_{n,i}$ の漸化式の展開から，
\begin{equation}
    \epsilon + v_{n,i} \geq \epsilon + \beta_2^k v_{n-k,i} \geq \beta_2^k(\epsilon + v_{n-k,i})\quad (0<\beta_2\le 1)
\end{equation}
したがって，
\begin{equation}
    \frac{g_{n-k,i}^2}{\epsilon + v_{n,i}}
    \leq \frac{1}{\beta_2^k} \cdot \frac{g_{n-k,i}^2}{\epsilon + v_{n-k,i}}
    = \frac{1}{\beta_2^k} U_{n-k,i}^2.
    \tag{A.16}
    \label{eq:ratio_bound}
\end{equation}
上式で $U_{n-k,i} = g_{n-k,i}/\sqrt{\epsilon+v_{n-k,i}}$ はモメンタムなし更新ベクトルの
第 $i$ 成分（$v_{n,i}$ ではなく $v_{n-k,i}$ で割っている点に注意）．

また，$F$ の $L$-滑らかさとJensenの不等式を用いると，
\begin{align}
    \|G_n - G_{n-k}\|_2^2
    &\leq L^2 \|x_{n-1} - x_{n-k-1}\|_2^2
     = L^2 \left\|\sum_{l=1}^{k} \alpha_{n-l} u_{n-l}\right\|_2^2
     \leq \alpha_n^2 L^2 k \sum_{l=1}^{k} \|u_{n-l}\|_2^2,
    \tag{A.17}
    \label{eq:smooth_bound}
\end{align}
ここで $\alpha_n$ が非減少であることを用いた。上式の証明は[\ref{pf:a17}]を参照。

\eqref{eq:ratio_bound} と \eqref{eq:smooth_bound} を \eqref{eq:B_bound} に代入して
インデックスを入れ替えると，
\begin{align}
    |B| &\leq
    \underbrace{
      \frac{\sqrt{1-\beta_1}\,\alpha_n^2 L^2}{4R}
      \left(\sum_{l=1}^{n-1} \|u_{n-l}\|_2^2\sum_{k=l}^{n-1} \beta_1^k{\color{red}{\sqrt{k}}}\right)
    }_{\text{$u_{n-l}$（モメンタム付き更新）に関わる項}}
    +
    \underbrace{
      \frac{R}{\sqrt{1-\beta_1}}
      \sum_{k=0}^{n-1} \left(\frac{\beta_1}{\beta_2}\right)^k \sqrt{k+1}\,\|U_{n-k}\|_2^2
    }_{\text{$U_{n-k}$（モメンタムなし更新）に関わる項}}.
    \tag{A.18}
    \label{eq:B_final}
\end{align}
ここで、${\color{red} k^2 \le k(k + 1)\implies \dfrac{k}{\sqrt{k + 1}} \le \sqrt{k}}$

\newpage
\noindent
\textbf{A 項の下界評価}

\noindent
\paragraph{A 項の内部構造の整理}

まず，$A$ 項における，ある固定した座標 $i \in [d]\ (d \in \mathbb{N})$ と
遅延インデックス $0 \le k \le n-1$ の寄与に注目する．
ここで簡潔さのために
\[
  t := n-k
\]
とおき，時刻 $t$ を「現在から $k$ ステップ前の時刻」とみなす．すると
\begin{equation}
    \mathbb{E}\!\left[\frac{G_{t,i}\, g_{t,i}}{\sqrt{\epsilon + v_{n,i}}}\right]
    = \mathbb{E}\!\left[
        \frac{\nabla_i F(x_{t-1})\, \nabla_i f_{t}(x_{t-1})}{\sqrt{\epsilon + v_{n,i}}}
      \right]
    \tag{A.19}
    \label{eq:Aterm_t}
\end{equation}
と書ける．
分母が $v_{t,i}$ ではなく $v_{n,i}$ である点に注意する．
ここが Lemma~5.1 との違いであり，分子と分母の相関をうまく分離するために
追加の工夫が必要になる．

以降は表記を簡潔にするため，インデックス $n,t,i$ を省略して
\begin{align*}
    G &= G_{t,i} = \nabla_i F(x_{t-1}), \\
    g &= g_{t,i} = \nabla_i f_{t}(x_{t-1}), \\
    \tilde{v} &= \tilde{v}_{n,k+1,i}, \\
    v &= v_{n,i}
\end{align*}
と書くことにする．

\medskip
\noindent
\textbf{$\tilde{v}_{n,k+1,i}$ の意味（「直近 $k+1$ 回」の期待値による置換）}

元の AdaGrad/Adam の更新では，
\[
  v_{n,i}
  = \beta_2 v_{n-1,i} + (1-\beta_2)\,g_{n,i}^2
\]
を繰り返すことで，各座標 $i$ について
\[
  v_{n,i}
  = \underbrace{\sum_{j=1}^{n-k-1} \beta_2^{\,n-j} g_{j,i}^2}_{\text{古い $n-k-1$ 回分の寄与}}
    \;+\;
    \underbrace{\sum_{j=n-k}^{n} \beta_2^{\,n-j} g_{j,i}^2}_{\text{直近 $k+1$ 回分の寄与}}
\]

ここで
\[
  \text{「直近 $k+1$ 回」} = \{j = n-k, n-k+1, \ldots, n\}
\]
の部分だけを「期待値で平滑化」したものが
\(\tilde{v}_{n,k+1,i}\) であり，
「$k+1$」という添え字は
「この $\tilde{v}$ が直近 $k+1$ 回分の寄与を
  条件付き期待値で置き換えた量である」
ことを明示している．

直感的には，
\[
  v_{n,i}
  = \text{古い部分} + \text{直近 $k+1$ 回の実現した勾配の二乗の加重和},
\]
\[
  \tilde{v}_{n,k+1,i}
  = \text{古い部分} + \text{直近 $k+1$ 回に対する期待される勾配二乗の加重和}
\]
と考えればよい．

簡単な例として：

$k = 0$（つまり $t = n$）の場合  
  直近 $k+1=1$ 回なので
  \[
    v_{n,i}
    = \sum_{j=1}^{n-1} \beta_2^{\,n-j} g_{j,i}^2
      + g_{n,i}^2,
  \]
  に対して
  \[
    \tilde{v}_{n,1,i}
    = \sum_{j=1}^{n-1} \beta_2^{\,n-j} g_{j,i}^2
      + \mathbb{E}_{n-1}[g_{n,i}^2],
  \]
  となる．ここで $\mathbb{E}_{n-1}[\cdot]$ は $f_1,\ldots,f_{n-1}$ による条件付き期待値である．

$k = 1$（つまり $t = n-1$）の場合  
  直近 $k+1=2$ 回なので
  \[
    v_{n,i}
    = \sum_{j=1}^{n-2} \beta_2^{\,n-j} g_{j,i}^2
      + \beta_2 g_{n-1,i}^2 + g_{n,i}^2,
  \]
  に対して
  \[
    \tilde{v}_{n,2,i}
    = \sum_{j=1}^{n-2} \beta_2^{\,n-j} g_{j,i}^2
      + \mathbb{E}_{n-2}\!\bigl[ \beta_2 g_{n-1,i}^2 + g_{n,i}^2 \bigr],
  \]
  となる．このように，添え字の $k+1$ は
  「$\tilde{v}$ の中で期待値に置き換えられているステップ数」
  をそのまま表している．

\medskip
\noindent
\textbf{$\Delta$ と $\bar{\Delta}$ の定義}

直近 $k+1$ 回の確率的勾配の「生の寄与」と，
その条件付き期待値を
\begin{equation}
    \Delta
    = \sum_{j=n-k}^{n} \beta_2^{\,n-j} g_{j,i}^2,
    \qquad
    \bar{\Delta}
    = \mathbb{E}_{n-k-1}[\Delta]
    \tag{A.20}
    \label{eq:Delta_barDelta}
\end{equation}
と定義する．
ここで $\mathbb{E}_{n-k-1}[\cdot]$ は，
過去のサンプル $f_1,\ldots,f_{n-k-1}$ によって生成された
フィルトレーションに関する条件付き期待値を表す．

$\Delta$ は「直近 $k+1$ 回（$j=n-k,\ldots,n$）で実際に観測された
  二乗勾配の加重和」．  
$\bar{\Delta}$ は「同じ区間に対する，$f_1,\ldots,f_{n-k-1}$ を固定したときの期待される二乗勾配の大きさ」（将来のサンプルに関する平均）．

例えば $k=0$ のときには
\[
  \Delta = g_{n,i}^2,\qquad
  \bar{\Delta} = \mathbb{E}_{n-1}[g_{n,i}^2]
\]
となり，「次の一回の勾配の二乗の条件付き期待値」を表す．

この定義から，$\tilde{v}$ と $v$ の差は
\[
  \tilde{v} - v = \bar{\Delta} - \Delta
\]
と書けることが直ちに分かる．

\medskip
\noindent
\textbf{A 項の分解}

\eqref{eq:Aterm_t} の分母を $\tilde{v}$ を用いて加減すると，
\begin{align}
    \mathbb{E}\!\left[\frac{Gg}{\sqrt{\epsilon+v}}\right]
    &= \mathbb{E}\!\left[\frac{Gg}{\sqrt{\epsilon+\tilde{v}}}\right]
     + \mathbb{E}\!\left[
        Gg\left(
          \frac{1}{\sqrt{\epsilon+v}} - \frac{1}{\sqrt{\epsilon+\tilde{v}}}
        \right)
      \right] \notag\\
    &= \mathbb{E}\!\left[\frac{G^2}{\sqrt{\epsilon+\tilde{v}}}\right]
     + \underbrace{\mathbb{E}\!\left[
         Gg \cdot
         \frac{\bar{\Delta} - \Delta}{
           \sqrt{\epsilon+v}\,\sqrt{\epsilon+\tilde{v}}\,
           \bigl(\sqrt{\epsilon+v}+\sqrt{\epsilon+\tilde{v}}\bigr)
         }
       \right]}_{\displaystyle C}.
    \tag{A.21}
    \label{eq:Aterm_decomp_Delta}
\end{align}
第 2 等号では，
$G$ と $\tilde{v}$ は $f_1,\ldots,f_{t-1}=f_1,\ldots,f_{n-k-1}$ のみに依存し，
一方で $g$ は $f_t=f_{n-k}\sim \text{i.i.d} $ にのみ依存するため，
条件付き期待値を用いると
\[
  \mathbb{E}_{t-1}\!\left[\frac{Gg}{\sqrt{\epsilon+\tilde{v}}}\right]
  = \frac{G}{\sqrt{\epsilon+\tilde{v}}}\,\mathbb{E}_{t-1}[g]
  = \frac{G^2}{\sqrt{\epsilon+\tilde{v}}}
\]
となることを用いた．
ここで最後の等号では，
$\mathbb{E}[\nabla f(x)] = \nabla F(x)$ という仮定を使っている．

\medskip
\noindent
\textbf{C 項の制御（$\kappa$ と $\rho$ への分解）}

$\sqrt{\epsilon+v} + \sqrt{\epsilon+\tilde{v}} \geq 
 \max(\sqrt{\epsilon+v},\,\sqrt{\epsilon+\tilde{v}})$
および $|\bar{\Delta} - \Delta| \leq \bar{\Delta} + \Delta$ から
\begin{align}
    |C|
    &= \left|
        Gg \cdot
        \frac{\bar{\Delta}-\Delta}{
          \sqrt{\epsilon+v}\,\sqrt{\epsilon+\tilde{v}}\,
          \bigl(\sqrt{\epsilon+v}+\sqrt{\epsilon+\tilde{v}}\bigr)
        }
      \right| \notag\\
    &= |Gg|\,
      \frac{|\bar{\Delta}-\Delta|}{
        \sqrt{\epsilon+v}\,\sqrt{\epsilon+\tilde{v}}\,
        \bigl(\sqrt{\epsilon+v}+\sqrt{\epsilon+\tilde{v}}\bigr)
      } \notag\\
    &\le
      |Gg|\,
      \frac{\bar{\Delta}+\Delta}{
        \sqrt{\epsilon+v}\,\sqrt{\epsilon+\tilde{v}}\,
        \bigl(\sqrt{\epsilon+v}+\sqrt{\epsilon+\tilde{v}}\bigr)
      }
      \label{eq:C_split_step1}\\
    &= \frac{|Gg|\,\bar{\Delta}}{
        \sqrt{\epsilon+v}\,\sqrt{\epsilon+\tilde{v}}\,
        \bigl(\sqrt{\epsilon+v}+\sqrt{\epsilon+\tilde{v}}\bigr)}
     + \frac{|Gg|\,\Delta}{
        \sqrt{\epsilon+v}\,\sqrt{\epsilon+\tilde{v}}\,
        \bigl(\sqrt{\epsilon+v}+\sqrt{\epsilon+\tilde{v}}\bigr)}
      \notag\\
    &\le
      \frac{|Gg|\,\bar{\Delta}}{
        \sqrt{\epsilon+v}\,\sqrt{\epsilon+\tilde{v}}\,
        \sqrt{\epsilon+\tilde{v}}}
     + \frac{|Gg|\,\Delta}{
        \sqrt{\epsilon+v}\,\sqrt{\epsilon+\tilde{v}}\,
        \sqrt{\epsilon+v}}
      \label{eq:C_split_step2}\\
    &= \frac{|Gg|\,\bar{\Delta}}{\sqrt{\epsilon+v}\,(\epsilon+\tilde{v})}
     + \frac{|Gg|\,\Delta}{(\epsilon+v)\,\sqrt{\epsilon+\tilde{v}}} \notag\\
    &= \underbrace{\frac{|Gg|\,\bar{\Delta}}{\sqrt{\epsilon+v}\,(\epsilon+\tilde{v})}}_{\displaystyle \kappa}
     + \underbrace{\frac{|Gg|\,\Delta}{(\epsilon+v)\,\sqrt{\epsilon+\tilde{v}}}}_{\displaystyle \rho}.
    \label{eq:kappa_rho_Delta}
\end{align}


\paragraph{$\kappa$ の評価}

ヤングの不等式 \eqref{eq:young} に対して
\begin{equation}
    \lambda = \frac{\sqrt{1-\beta_1}\,\sqrt{\epsilon+\tilde{v}}}{2},\quad
    x = \frac{|G|}{\sqrt{\epsilon+\tilde{v}}},\quad
    y = \frac{|g|\,\bar{\Delta}}{\sqrt{\epsilon+\tilde{v}}\,\sqrt{\epsilon+v}}
\end{equation}
と選ぶと，
\begin{align}
    \kappa
    &= \frac{|Gg|\,\bar{\Delta}}{\sqrt{\epsilon+v}\,(\epsilon+\tilde{v})} \notag\\
    &= \left(\frac{|G|}{\sqrt{\epsilon+\tilde{v}}}\right)
       \left(
         \frac{|g|\,\bar{\Delta}}
              {\sqrt{\epsilon+v}\,\sqrt{\epsilon+\tilde{v}}}
       \right)
       \label{eq:kappa_young_step1}
\end{align}
Young の不等式
\[
  xy \le \frac{\lambda}{2}x^2 + \frac{1}{2\lambda}y^2
\]
を
\[
  \lambda = \frac{\sqrt{1-\beta_1}\,\sqrt{\epsilon+\tilde{v}}}{2},\qquad
  x = \frac{|G|}{\sqrt{\epsilon+\tilde{v}}},\qquad
  y = \frac{|g|\,\bar{\Delta}}
           {\sqrt{\epsilon+v}\,\sqrt{\epsilon+\tilde{v}}}
\]
として \eqref{eq:kappa_young_step1} に適用すると，
\begin{align}
    \kappa
    &\le
    \frac{1}{2}
    \cdot
    \frac{\sqrt{1-\beta_1}\,\sqrt{\epsilon+\tilde{v}}}{2}
    \cdot
    \frac{G^2}{\epsilon+\tilde{v}}
    +
    \frac{1}{2}
    \cdot
    \frac{2}{\sqrt{1-\beta_1}\,\sqrt{\epsilon+\tilde{v}}}
    \cdot
    \frac{g^2\bar{\Delta}^{\,2}}
         {(\epsilon+v)(\epsilon+\tilde{v})}
    \notag\\
    &= \frac{\sqrt{1-\beta_1}}{4}\,
       \frac{G^2}{\sqrt{\epsilon+\tilde{v}}}
    + \frac{1}{\sqrt{1-\beta_1}}\,
      \frac{g^2\bar{\Delta}^{\,2}}
           {(\epsilon+\tilde{v})^{3/2}(\epsilon+v)}
    \label{eq:kappa_young_step2}
\end{align}
最後に，$0 < 1-\beta_1 \le 1$ であるから
\[
  \sqrt{1-\beta_1} \le 1
\]
を用いると，
\begin{equation}
    \kappa \leq \frac{G^2}{4\sqrt{\epsilon+\tilde{v}}}
    + \frac{1}{\sqrt{1-\beta_1}}\,
      \frac{g^2 \bar{\Delta}^{\,2}}{(\epsilon+\tilde{v})^{3/2}(\epsilon+v)}.
    \label{eq:kappa_bound_detail}
\end{equation}

さらに $\epsilon + \tilde{v} \geq \bar{\Delta}$（$\tilde{v}$ の定義より $\tilde{v}$ は
直近 $k+1$ 回の項を条件付き期待値で置き換えたものであり，
その分は $\bar{\Delta}$ で下から抑えられる）を用いて
条件付き期待値 $\mathbb{E}_{n-k-1}[\cdot]$ をとると
\begin{equation}
    \kappa \leq \frac{G^2}{4\sqrt{\epsilon+\tilde{v}}}
    + \frac{1}{\sqrt{1-\beta_1}}\,
      \frac{g^2 \bar{\Delta}^{\,2}}{(\epsilon+\tilde{v})^{3/2}(\epsilon+v)}.
    \label{eq:kappa_before_conditional}
\end{equation}
\eqref{eq:kappa_before_conditional} に対し、両辺に条件付き期待値をとると，
$\mathbb{E}_{n-k-1}[\cdot]$ をとると，
\begin{align}
    \mathbb{E}_{n-k-1}[\kappa]
    &\leq
    \mathbb{E}_{n-k-1}\!\left[
      \frac{G^2}{4\sqrt{\epsilon+\tilde{v}}}
    \right]
    + \frac{1}{\sqrt{1-\beta_1}}\,
      \mathbb{E}_{n-k-1}\!\left[
        \frac{g^2 \bar{\Delta}^{\,2}}
             {(\epsilon+\tilde{v})^{3/2}(\epsilon+v)}
      \right]
    \label{eq:kappa_cond_step1}\\
    &= \frac{G^2}{4\sqrt{\epsilon+\tilde{v}}}
    + \frac{1}{\sqrt{1-\beta_1}}\,
      \frac{\bar{\Delta}^{\,2}}{(\epsilon+\tilde{v})^{3/2}}
      \mathbb{E}_{n-k-1}\!\left[
        \frac{g^2}{\epsilon+v}
      \right].
    \label{eq:kappa_cond_step2}
\end{align}
ここで，$G=\nabla_iF(x_{n-k-1})$ は $x_{n-k-1}$ のみの関数であり，
また $\tilde v$ と $\bar\Delta=\mathbb{E}_{n-k-1}[\Delta]$ も
$f_1,\dots,f_{n-k-1}$ に関して可測であるから，
いずれも $\mathbb{E}_{n-k-1}[\cdot]$ の外に出すことができる．

さらに，$\tilde v$ の定義より
\[
  \epsilon+\tilde v \ge \bar\Delta
\]
であるので，
\begin{equation}
    \frac{\bar{\Delta}^{\,2}}{(\epsilon+\tilde{v})^{3/2}}
    =
    \frac{\bar{\Delta}}{\sqrt{\epsilon+\tilde{v}}}
    \cdot
    \frac{\bar{\Delta}}{\epsilon+\tilde{v}}
    \le
    \frac{\bar{\Delta}}{\sqrt{\epsilon+\tilde{v}}}.
    \label{eq:kappa_cond_step3}
\end{equation}
これを \eqref{eq:kappa_cond_step2} に代入すると，
\begin{equation}
    \mathbb{E}_{n-k-1}[\kappa]
    \leq \frac{G^2}{4\sqrt{\epsilon+\tilde{v}}}
    + \frac{1}{\sqrt{1-\beta_1}}\,
      \frac{\bar{\Delta}}{\sqrt{\epsilon+\tilde{v}}}
      \mathbb{E}_{n-k-1}\!\left[\frac{g^2}{\epsilon+v}\right].
    \tag{A.22}
    \label{eq:kappa_bound_Delta}
\end{equation}

ここで $\sqrt{\mathbb{E}_{n-k-1}[g^2]} \leq \sqrt{\epsilon+\tilde{v}}$ と
$\sqrt{\mathbb{E}_{n-k-1}[g^2]} \leq R$から，
\[
  \frac{\bar{\Delta}}{\sqrt{\epsilon+\tilde{v}}}
  \leq \frac{R}{\sqrt{1-\beta_1}}
\]
\textbf{（$R$ は確率的勾配の $\ell_\infty$ ノルムの上界）}
となるので，
\begin{equation}
    \mathbb{E}_{n-k-1}[\kappa]
    \leq \frac{G^2}{4\sqrt{\epsilon+\tilde{v}}}
    + \frac{R}{\sqrt{1-\beta_1}}
      \mathbb{E}_{n-k-1}\!\left[\frac{g^2}{\epsilon+v}\right].
    \tag{A.22'}
    \label{eq:kappa_simplified_Delta}
\end{equation}
\medskip
\noindent
\paragraph{$G,\tilde{v},\bar{\Delta}$ が $f_1,\dots,f_{n-k-1}$ に関して可測である理由}\ \\
\noindent
ここで用いている条件付き期待値
$\mathbb{E}_{n-k-1}[\cdot]$ は，
$f_1,\dots,f_{n-k-1}$ によって生成される情報の下での期待値である．
したがって，ある量が $f_1,\dots,f_{n-k-1}$ に関して可測であるとは，
それが「過去のサンプル $f_1,\dots,f_{n-k-1}$ だけから一意に決まる」
ことを意味する．

まず，
\[
  G = \nabla_i F(x_{n-k-1})
\]
について考える．
反復点 $x_{n-k-1}$ は，初期値 $x_0$ と，
それまでに観測された確率的勾配
\[
  \nabla f_1(x_0),\ \nabla f_2(x_1),\ \dots,\ \nabla f_{n-k-1}(x_{n-k-2})
\]
を用いた更新則の繰り返しによって定まる．
したがって $x_{n-k-1}$ は $f_1,\dots,f_{n-k-1}$ のみの関数であり，
よって $G=\nabla_i F(x_{n-k-1})$ も
$f_1,\dots,f_{n-k-1}$ に関して可測である．

次に，
\[
  \tilde{v}=\tilde{v}_{n,k+1,i}
\]
について見る．
$\tilde{v}$ は，$v_{n,i}$ のうち
古い部分
\[
  \sum_{j=1}^{n-k-1}\beta_2^{\,n-j} g_{j,i}^2
\]
をそのまま残し，
直近 $k+1$ 回に対応する部分
\[
  \sum_{j=n-k}^{n}\beta_2^{\,n-j} g_{j,i}^2
\]
を条件付き期待値で置き換えた量として定義されている．
前者は明らかに $f_1,\dots,f_{n-k-1}$ のみから決まり，
後者も「$f_1,\dots,f_{n-k-1}$ を固定した下での期待値」であるから，
その値は過去の情報だけで決まる．
したがって，$\tilde{v}$ 全体も
$f_1,\dots,f_{n-k-1}$ に関して可測である．

同様に，
\[
  \bar{\Delta}
  = \mathbb{E}_{n-k-1}[\Delta]
\]
も
$f_1,\dots,f_{n-k-1}$ に関して可測である．
実際，条件付き期待値の一般論として，
$\mathbb{E}_{n-k-1}[\Delta]$ は
$f_1,\dots,f_{n-k-1}$ に関する情報だけに依存する量として定義される．
直感的には，$\Delta$ 自体は
$f_{n-k},\dots,f_n$ のランダム性を含むが，
$\bar{\Delta}$ はそれら将来のランダム性を平均化してしまったものであり，
過去の情報を与えればその値は確定する．

以上より，
$G,\tilde{v},\bar{\Delta}$ はいずれも
$f_1,\dots,f_{n-k-1}$ に関して可測である．
そのため，$\mathbb{E}_{n-k-1}[\cdot]$ をとる際には，
これらは条件付き期待値の外に出すことができる．
一方で，
\[
  g = g_{n-k,i} = \nabla_i f_{n-k}(x_{n-k-1})
\]
は新しいサンプル $f_{n-k}$ に依存するため，
一般には $f_1,\dots,f_{n-k-1}$ に関して可測ではない．
この違いが，
条件付き期待値の中で $G,\tilde{v},\bar{\Delta}$ を定数として扱い，
$g$ のみを平均化することを可能にしている．


\paragraph{$\rho$ の評価}

同様に \eqref{eq:young} に
\begin{equation}
    \lambda = \frac{\sqrt{1-\beta_1}\,\sqrt{\epsilon+\tilde{v}}}{2\bar{\Delta}},\quad
    x = \frac{|G|\sqrt{\Delta}}{\sqrt{\epsilon+\tilde{v}}},\quad
    y = \frac{\sqrt{\Delta}\,|g|}{\epsilon+v}
\end{equation}
と代入すると，
\begin{equation}
    \rho \leq \frac{G^2}{4\sqrt{\epsilon+\tilde{v}}} \cdot \frac{\Delta}{\bar{\Delta}}
    + \frac{1}{\sqrt{1-\beta_1}}\,
      \frac{\bar{\Delta}}{\sqrt{\epsilon+\tilde{v}}}
      \cdot \frac{g^2\Delta}{(\epsilon+v)^2}.
    \tag{A.23}
    \label{eq:rho_bound_Delta}
\end{equation}
$\epsilon + v \geq \Delta$ および
\(\mathbb{E}_{n-k-1}[\Delta/\bar{\Delta}] = 1\)（\(\bar{\Delta} = \mathbb{E}_{n-k-1}[\Delta]\) の定義）を用いて
条件付き期待値をとると，
\begin{equation}
    \mathbb{E}_{n-k-1}[\rho]
    \leq \frac{G^2}{4\sqrt{\epsilon+\tilde{v}}}
    + \frac{R}{\sqrt{1-\beta_1}}
      \mathbb{E}_{n-k-1}\!\left[\frac{g^2}{\epsilon+v}\right].
    \tag{A.24}
    \label{eq:rho_simplified_Delta}
\end{equation}
（もし $\bar{\Delta} = 0$ ならば，$\Delta = 0$ a.s. となるので $\rho = 0$ が成り立つ．）

\paragraph{$C$ の上界のまとめ}

\eqref{eq:kappa_simplified_Delta} と \eqref{eq:rho_simplified_Delta} を加えると，
\begin{equation}
    \mathbb{E}_{n-k-1}[|C|]
    \leq \frac{G^2}{2\sqrt{\epsilon+\tilde{v}}}
    + \frac{2R}{\sqrt{1-\beta_1}}
      \mathbb{E}_{n-k-1}\!\left[\frac{g^2}{\epsilon+v}\right].
    \tag{A.25}
    \label{eq:C_bound_Delta}
\end{equation}
さらに，\(\bar{\Delta} \leq \epsilon+\tilde{v}\) および
\(\bar{\Delta} \leq (k+1)R^2\)から勾配の絶対値が $R$ で抑えられるため，
直近 $k+1$ 回の和も \(\sqrt{k+1}\,R\) のオーダーで抑えられる）を用いて
インデックスを元に戻すと，
\begin{equation}
    \mathbb{E}_{n-k-1}[|C|]
    \leq \frac{G_{t,i}^2}{2\sqrt{\epsilon+\tilde{v}_{n,k+1,i}}}
    + \frac{2R\sqrt{k+1}}{\sqrt{1-\beta_1}}
      \mathbb{E}_{n-k-1}\!\left[\frac{g_{t,i}^2}{\epsilon+v_{n,i}}\right],
    \qquad(t = n-k).
    \tag{A.26}
    \label{eq:C_final_Delta}
\end{equation}
最後に完全期待値をとり，
\(\epsilon + v_{n,i} \geq \beta_2^k(\epsilon+v_{t,i})\)（\(t=n-k\)）を用いると，
\begin{equation}
    \mathbb{E}[|C|]
    \leq \frac{1}{2}\mathbb{E}\!\left[
      \frac{G_{t,i}^2}{\sqrt{\epsilon+\tilde{v}_{n,k+1,i}}}
    \right]
    + \frac{2R\sqrt{k+1}}{\sqrt{1-\beta_1}\,\beta_2^k}
      \mathbb{E}\!\left[\frac{g_{t,i}^2}{\epsilon+v_{t,i}}\right],
    \qquad(t = n-k).
    \tag{A.27}
    \label{eq:C_expectation_Delta}
\end{equation}

\bigskip
\noindent
\textbf{A 項の下界のまとめ}

\eqref{eq:C_expectation_Delta} を
\eqref{eq:Aterm_decomp_Delta} に代入し，
全ての座標 $i \in [d]$ と全ての遅延 $k = 0, \ldots, n-1$ について
$\beta_1^k$ の重みをつけて和をとると，
\begin{align}
    \mathbb{E}[A]
    &\geq
    \frac{1}{2} \sum_{i \in [d]} \sum_{k=0}^{n-1} \beta_1^k
    \mathbb{E}\!\left[
      \frac{G_{n-k,i}^2}{\sqrt{\epsilon+\tilde{v}_{n,k+1,i}}}
    \right]
    - \frac{2R}{\sqrt{1-\beta_1}}
      \sum_{i \in [d]} \sum_{k=0}^{n-1}
      \left(\frac{\beta_1}{\beta_2}\right)^k \sqrt{k+1}\,
      \mathbb{E}\!\left[\|U_{n-k}\|_2^2\right].
    \tag{A.28}
    \label{eq:A_lower_Delta}
\end{align}

\bigskip
\noindent
\textbf{最終的な不等式の確定}

\eqref{eq:A_lower_Delta}（$A$ の下界）と
\eqref{eq:B_final}（$|B|$ の上界）を
\eqref{eq:AB_decomp} に代入する．
\[
  \mathbb{E}\Bigl[\sum_i G_{n,i} m_{n,i}/\sqrt{\epsilon+v_{n,i}}\Bigr]
  = \mathbb{E}[A] - \mathbb{E}[|B|]
\]
であることから，
\begin{align}
    \mathbb{E}\!\left[\sum_{i \in [d]} G_{n,i}
      \frac{m_{n,i}}{\sqrt{\epsilon+v_{n,i}}}\right]
    &\geq
    \frac{1}{2} \sum_{i \in [d]} \sum_{k=0}^{n-1} \beta_1^k
    \mathbb{E}\!\left[
      \frac{G_{n-k,i}^2}{\sqrt{\epsilon+\tilde{v}_{n,k+1,i}}}
    \right] \notag\\
    &\quad
    - \frac{\alpha_n^2 L^2}{4R\sqrt{1-\beta_1}}
      \sum_{l=1}^{n-1} \|u_{n-l}\|_2^2
      \sum_{k=l}^{n-1} \frac{\beta_1^k}{\sqrt{k}} \notag\\
    &\quad
    - \frac{3R}{\sqrt{1-\beta_1}}
      \sum_{k=0}^{n-1}
      \left(\frac{\beta_1}{\beta_2}\right)^k \sqrt{k+1}\,\|U_{n-k}\|_2^2,
\end{align}
これが \eqref{eq:lemmaA1} に相当する不等式である．
\end{proof}

\subsubsection*{時間インデックスと各変数の関係}

ここでは，証明に登場する時間インデックス $n, k, t$ と，
$G_{t,i}, g_{t,i}, v_{n,i}, \tilde{v}_{n,k+1,i}, \Delta, \bar{\Delta}$ の関係を直感的に整理する．

\paragraph{基本となる時間の軸}

\begin{itemize}
  \item $n$：現在の反復回数（$n$ ステップ目）．
  \item $k$：どれだけ過去の勾配を見るかを表す「遅延インデックス」．
  \item $t = n-k$：$k$ ステップ前の時刻．
\end{itemize}

したがって，$k=0$ なら $t=n$（現在時刻），$k=1$ なら $t=n-1$（1 ステップ前），
一般に $t$ は「今から $k$ ステップさかのぼった時刻」を意味する．

\paragraph{勾配に関する変数 $G_{t,i}, g_{t,i}$}

\begin{itemize}
  \item $G_{t,i} = \nabla_i F(x_{t-1})$：決定論的目的関数 $F$ の
    「時刻 $t-1$ の点 $x_{t-1}$ における $i$ 成分の真の勾配」．
    これは $f_1,\ldots,f_{t-1}$ までの情報だけに依存する．
  \item $g_{t,i} = \nabla_i f_t(x_{t-1})$：
    「時刻 $t$ にサンプリングされた損失 $f_t$ に対する確率的勾配」．
    これは新しい乱数 $f_t$ に依存し，$t$ 以降のランダム性を含む．
\end{itemize}

したがって，$t-1$ までの情報（すなわち $f_1,\ldots,f_{t-1}$）を固定すると，
$G_{t,i}$ は定数，$g_{t,i}$ は「これから観測される乱数」によるランダム変数になる．

\paragraph{累積二乗勾配 $v_{n,i}$ とその平滑化 $\tilde{v}_{n,k+1,i}$}

\begin{itemize}
  \item $v_{n,i}$：$\beta_2$ による指数移動平均で集計した
    「時刻 $1$ から $n$ までのすべての二乗勾配の寄与」の和．
    過去すべての $g_{j,i}^2$（$1 \le j \le n$）が，重み $\beta_2^{\,n-j}$ で入っている．
  \item $\tilde{v}_{n,k+1,i}$：
    $v_{n,i}$ を
    \[
      v_{n,i}
      =
      \underbrace{\sum_{j=1}^{n-k-1} \beta_2^{\,n-j} g_{j,i}^2}_{\text{古い部分}}
      +
      \underbrace{\sum_{j=n-k}^{n} \beta_2^{\,n-j} g_{j,i}^2}_{\text{直近 $k+1$ 回分}}
    \]
    と分解したときに，
    「古い部分」はそのまま，
    「直近 $k+1$ 回分」のみを条件付き期待値で置き換えたもの．
\end{itemize}

添え字の $k+1$ は，
「$\tilde{v}$ の中で，期待値に置き換えられているステップ数が
  ちょうど直近 $k+1$ 回である」ことを明示している．
例として：

\begin{itemize}
  \item $k=0$ のとき（$t=n$）は「直近 $1$ 回」だけを期待値で置き換えるので，
    \[
      \tilde{v}_{n,1,i}
      = \sum_{j=1}^{n-1} \beta_2^{\,n-j} g_{j,i}^2
        + \mathbb{E}_{n-1}[g_{n,i}^2] .
    \]
  \item $k=1$ のとき（$t=n-1$）は「直近 $2$ 回」分を期待値で置き換えるので，
    \[
      \tilde{v}_{n,2,i}
      = \sum_{j=1}^{n-2} \beta_2^{\,n-j} g_{j,i}^2
        + \mathbb{E}_{n-2}\!\bigl[\beta_2 g_{n-1,i}^2 + g_{n,i}^2\bigr] .
    \]
\end{itemize}

このように，
「$n$ は現在の時刻，$k$ はどこまでさかのぼるか，
 $k+1$ は期待値で平滑化される直近ステップ数」
という役割分担になっている．

\paragraph{$\Delta$ と $\bar{\Delta}$ の時間的意味}

\begin{itemize}
  \item $\Delta = \displaystyle\sum_{j=n-k}^{n} \beta_2^{\,n-j} g_{j,i}^2$：
    時刻 $t=n-k$ から現在 $n$ までの
    「直近 $k+1$ 回の実際に観測された二乗勾配の加重和」．
  \item $\bar{\Delta} = \mathbb{E}_{n-k-1}[\Delta]$：
    $f_1,\ldots,f_{n-k-1}$ までを固定したときに，
    「これから現れる $f_{n-k},\ldots,f_n$ に対して
      期待される直近 $k+1$ 回分の二乗勾配の大きさ」．
\end{itemize}

\noindent
したがって，
\[
  v_{n,i}
  = \text{古い部分} + \Delta, \qquad
  \tilde{v}_{n,k+1,i}
  = \text{古い部分} + \bar{\Delta},
\]
となり，
\[
  \tilde{v}_{n,k+1,i} - v_{n,i} = \bar{\Delta} - \Delta
\]
が成り立つ．
時間の流れとしては，「$1,\ldots,n-k-1$ の寄与はすでに確定しており，
   $n-k,\ldots,n$ に対応する部分だけを
   実現値 $\Delta$ から期待値 $\bar{\Delta}$ に入れ替えている」
と理解すればよい．

\begin{itemize}
  \item $t = n-k$：$k$ ステップ前の時刻（$G_{t,i}, g_{t,i}$ の時間）．
  \item $n$：累積量 $v_{n,i}, \tilde{v}_{n,k+1,i}$ の評価時刻．
  \item $k+1$：$v_{n,i}$ のうち，$\Delta,\bar{\Delta}$ として扱っている「直近の窓幅」．
\end{itemize}
この対応関係を頭に置くと，A 項の分解で出てくる
$\mathbb{E}_{n-k-1}[\cdot]$ や $\tilde{v}_{n,k+1,i}$ の役割が
かなり追いやすくなる．

\subsection{ヤング不等式の証明}\label{pf:youngs_ineq}

\begin{lemma}[ヤング不等式]
\label{lemma:young}
任意の $\lambda > 0$ および $x, y \in \mathbb{R}$ に対して，次の不等式が成り立つ：
\begin{equation}
    xy \leq \frac{\lambda}{2} x^2 + \frac{y^2}{2\lambda}.
    \tag{A.13}
    \label{eq:young}
\end{equation}
\end{lemma}

\begin{proof}

\paragraph{非負性の利用}\ \\

任意の実数 $a \in \mathbb{R}$ に対して，二乗は常に非負であるから，
\begin{equation}
    \left(\sqrt{\lambda}\,x - \frac{y}{\sqrt{\lambda}}\right)^2 \geq 0.
    \label{eq:sq_nonneg}
\end{equation}
ここで $\lambda > 0$ であるから $\sqrt{\lambda}$ は実数として定義される．

\paragraph{式の展開}\ \\

\eqref{eq:sq_nonneg} の左辺を展開すると，
\begin{align}
    \left(\sqrt{\lambda}\,x - \frac{y}{\sqrt{\lambda}}\right)^2
    &= \left(\sqrt{\lambda}\,x\right)^2
       - 2 \cdot \sqrt{\lambda}\,x \cdot \frac{y}{\sqrt{\lambda}}
       + \left(\frac{y}{\sqrt{\lambda}}\right)^2 \notag \\
    &= \lambda x^2 - 2xy + \frac{y^2}{\lambda}.
    \label{eq:expand}
\end{align}

\paragraph{不等式の変形}\ \\

\eqref{eq:sq_nonneg} と \eqref{eq:expand} を合わせると，
\begin{equation}
    \lambda x^2 - 2xy + \frac{y^2}{\lambda} \geq 0.
\end{equation}
両辺を整理して $2xy$ を右辺に移項すると，
\begin{equation}
    \lambda x^2 + \frac{y^2}{\lambda} \geq 2xy.
\end{equation}
両辺を $2$ で割ると，
\begin{equation}
    \frac{\lambda}{2} x^2 + \frac{y^2}{2\lambda} \geq xy.
\end{equation}
すなわち，
\begin{equation}
    xy \leq \frac{\lambda}{2} x^2 + \frac{y^2}{2\lambda}
\end{equation}

\end{proof}

\begin{rem}[等号条件]
等号は \eqref{eq:sq_nonneg} において等号が成立するとき，すなわち
\begin{equation}
    \sqrt{\lambda}\,x = \frac{y}{\sqrt{\lambda}}
    \quad \Longleftrightarrow \quad
    y = \lambda x
\end{equation}
が成立するときに限る．
\end{rem}

\begin{rem}[AM-GM 不等式との関係]
$a = \sqrt{\lambda}\,|x|$，$b = |y|/\sqrt{\lambda}$ とおくと，
\eqref{eq:young} は相加・相乗平均不等式（AM-GM 不等式）
\begin{equation}
    \frac{a^2 + b^2}{2} \geq ab
    \quad (a, b \geq 0)
\end{equation}
の直接的な言い換えに過ぎない．実際，
\begin{equation}
    \frac{a^2 + b^2}{2}
    = \frac{\lambda x^2}{2} + \frac{y^2}{2\lambda}
    \geq ab
    = \sqrt{\lambda}\,|x| \cdot \frac{|y|}{\sqrt{\lambda}}
    = |xy|
    \geq xy.
\end{equation}
\end{rem}

\begin{rem}[スケーリングパラメータ $\lambda$ の役割]
$\lambda$ は $x^2$ と $y^2$ の係数のバランスを自由に調整するパラメータである．
\begin{itemize}
    \item $\lambda \to \infty$ のとき：$\lambda x^2 / 2$ が支配的になり，
          $x$ の大きさをより重く評価する．
    \item $\lambda \to 0^+$ のとき：$y^2 / (2\lambda)$ が支配的になり，
          $y$ の大きさをより重く評価する．
    \item $\lambda = 1$ のとき：標準的な形 $xy \leq (x^2 + y^2)/2$ が得られる．
\end{itemize}
Lemma~A.1 の証明では，$\lambda$ を具体的に
\begin{equation}
    \lambda = \frac{\sqrt{1-\beta_1}\,\sqrt{\epsilon + \tilde{v}}}{2}
\end{equation}
などと選ぶことで，現れる項のバランスを取り，
主要項と誤差項を適切に分離している．
\end{rem}

\subsection{式 [A.17] 右側の不等式の証明}\label{pf:a17}
\begin{reidai}{式 [A.17] 右側の不等式の証明}{dd}
$$
    L^2 \left\|\sum_{l=1}^{k} \alpha_{n-l} u_{n-l}\right\|_2^2\leq \alpha_n^2 L^2 k \sum_{l=1}^{k} \|u_{n-l}\|_2^2 
$$
証明せよ

\end{reidai}
示すべき不等式は，$(\alpha_j)_{j \in \mathbb{N}^*}$ が非減少列であるとき，
\begin{equation}
    \left\|\sum_{l=1}^{k} \alpha_{n-l}\, u_{n-l}\right\|_2^2
    \leq \alpha_n^2\, k \sum_{l=1}^{k} \|u_{n-l}\|_2^2
    \label{eq:target}
\end{equation}
が成立することである．

\begin{proof}

\paragraph{Step 1：コーシー・シュワルツ不等式の適用}

ベクトルの内積に関するコーシー・シュワルツ不等式
\begin{equation}
    \left\|\sum_{l=1}^{k} a_l\right\|_2^2
    \leq k \sum_{l=1}^{k} \|a_l\|_2^2
    \label{eq:CS_vec}
\end{equation}
は，$a_l \in \mathbb{R}^d$ に対して次のように導かれる．
実際，コーシー・シュワルツ不等式
\begin{equation}
    \left(\sum_{l=1}^{k} \langle a_l,\, v \rangle\right)^2
    \leq k \sum_{l=1}^{k} \langle a_l,\, v \rangle^2
    \leq k \sum_{l=1}^{k} \|a_l\|_2^2 \|v\|_2^2
    \quad \forall\, v \in \mathbb{R}^d,\; \|v\|_2 = 1
\end{equation}
から \eqref{eq:CS_vec} が成立する．
これを $a_l = \alpha_{n-l}\, u_{n-l}$ に適用すると，
\begin{equation}
    \left\|\sum_{l=1}^{k} \alpha_{n-l}\, u_{n-l}\right\|_2^2
    \leq k \sum_{l=1}^{k} \|\alpha_{n-l}\, u_{n-l}\|_2^2
    = k \sum_{l=1}^{k} \alpha_{n-l}^2\, \|u_{n-l}\|_2^2.
    \label{eq:after_CS}
\end{equation}
\end{proof}

\subsection{条件付き期待値 $\mathbb{E}_{n-k-1}[\cdot]$ の具体的な意味}

\paragraph{前提となる変数の定義（証明中の省略記法）}

証明中では，固定した $n \in \mathbb{N}^*$，$i \in [d]$，$0 \leq k \leq n-1$ に対して，
以下の省略記法を用いる（論文中の式 (A.19) 直後の段落）：
\begin{align*}
    G &= G_{n-k,i} = \nabla_i F(x_{n-k-1})
    &&\text{【時刻 $n-k$ における決定論的勾配の第 $i$ 成分】}\\
    g &= g_{n-k,i} = \nabla_i f_{n-k}(x_{n-k-1})
    &&\text{【時刻 $n-k$ における確率的勾配の第 $i$ 成分】}\\
    v &= v_{n,i}
    &&\text{【現在時刻 $n$ における二乗勾配の指数移動平均】}\\
    \tilde{v} &= \tilde{v}_{n,k+1,i}
    &&\text{【$v_{n,i}$ のうち，最新 $k+1$ 個の確率的寄与を期待値に置換したもの】}
\end{align*}


\medskip
\paragraph{$\tilde{v}_{n,k+1,i}$ の定義（式 (A.7)）}

$v_{n,i}$ の漸化式を展開すると，
\begin{equation}
    v_{n,i} = \sum_{j=1}^{n} \beta_2^{n-j} g_{j,i}^2
    \label{eq:v_expand}
\end{equation}
と表せる（$v_0 = 0$ と初期化しているため）．
これを「過去の全寄与の和」と「直近 $k+1$ 個の寄与」に分けると，
\begin{equation}
    v_{n,i}
    = \underbrace{\beta_2^{k+1} v_{n-k-1,i}}_{\text{時刻 $n-k-1$ 以前の寄与}}
    + \underbrace{\sum_{j=n-k}^{n} \beta_2^{n-j} g_{j,i}^2}_{\text{直近 $k+1$ 個の寄与（$j = n-k, \ldots, n$）}}.
    \label{eq:v_split}
\end{equation}
ここで $\tilde{v}_{n,k+1,i}$ は，上式の「直近 $k+1$ 個の確率的寄与」の部分だけを
$\mathbb{E}_{n-k-1}[\,\cdot\,]$（$f_1, \ldots, f_{n-k-1}$ を所与とした条件付き期待値）で
置き換えたものとして定義される（論文 (A.7)）：
\begin{align}
    \tilde{v}_{n,k+1,i}
    &= \beta_2^k v_{n-k,i}
      + \mathbb{E}_{n-k-1}\!\left[\sum_{j=n-k}^{n} \beta_2^{n-j} g_{j,i}^2\right] \notag\\
    &= \beta_2^k v_{n-k,i}
      + \sum_{j=n-k}^{n} \beta_2^{n-j} \mathbb{E}_{n-k-1}\!\left[g_{j,i}^2\right].
    \label{eq:vtilde_def}
\end{align}
すなわち，$\tilde{v}$ は $v_{n,i}$ の「条件付き期待値版」であり，
決定論的な量（$f_1,\ldots,f_{n-k-1}$ を所与にすると定数）である．


\bigskip
\paragraph{$r^2$ の条件付き期待値としての具体的な意味}

$r^2 = \mathbb{E}_{n-k-1}[\delta^2]$ を明示的に計算すると，
\begin{align}
    r^2
    &= \mathbb{E}_{n-k-1}\!\left[\sum_{j=n-k}^{n} \beta_2^{n-j} g_{j,i}^2\right] \notag\\
    &= \sum_{j=n-k}^{n} \beta_2^{n-j}\, \mathbb{E}_{n-k-1}\!\left[g_{j,i}^2\right] \notag\\
    &= \sum_{j=n-k}^{n} \beta_2^{n-j}\,
       \mathbb{E}_{n-k-1}\!\left[\bigl(\nabla_i f_j(x_{j-1})\bigr)^2\right].
    \label{eq:r2_explicit}
\end{align}
\noindent
\textbf{【条件付き期待値 $\mathbb{E}_{n-k-1}[\,\cdot\,]$ の意味】}
$\mathbb{E}_{n-k-1}[\,\cdot\,]$ は「$f_1, f_2, \ldots, f_{n-k-1}$ を観測済みとして，
それ以降の確率性 $f_{n-k}, f_{n-k+1}, \ldots$ に関する期待値」を意味する．



\noindent
$g$ と $\tilde{v}$ は $\mathbb{E}_{n-k-1}$ のもとで互いに独立なので，
第1項 $\mathbb{E}[G^2/\sqrt{\epsilon+\tilde{v}}]$ は証明の主要項（下降方向の寄与）を表す．
$C$ 項は $r^2 - \delta^2 = \tilde{v} - v$ によるずれ（確率的変動）を表し，
これをヤング不等式と $r \leq \sqrt{k+1}\,R$（仮定 (7)）によって制御することが，
証明の核心となっている．
